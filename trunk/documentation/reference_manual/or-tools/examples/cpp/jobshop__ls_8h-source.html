00014 // This model implements a simple jobshop problem.
00015 //
00016 // A jobshop is a standard scheduling problem where you must schedule a
00017 // set of jobs on a set of machines.  Each job is a sequence of tasks
00018 // (a task can only start when the preceding task finished), each of
00019 // which occupies a single specific machine during a specific
00020 // duration. Therefore, a job is simply given by a sequence of pairs
00021 // (machine id, duration).
00022 
00023 // The objective is to minimize the 'makespan', which is the duration
00024 // between the start of the first task (across all machines) and the
00025 // completion of the last task (across all machines).
00026 //
00027 // This will be modelled by sets of intervals variables (see class
00028 // IntervalVar in constraint_solver/constraint_solver.h), one per
00029 // task, representing the [start_time, end_time] of the task.  Tasks
00030 // in the same job will be linked by precedence constraints.  Tasks on
00031 // the same machine will be covered by Sequence constraints.
00032 //
00033 // Search will be implemented as local search on the sequence variables.
00034

00052 class SwapIntervals : public SequenceVarLocalSearchOperator {
00053  public:
00054   SwapIntervals(const SequenceVar* const* vars, int size)
00055       : SequenceVarLocalSearchOperator(vars, size),
00056         current_var_(-1),
00057         current_first_(-1),
00058         current_second_(-1) {}
00059 
00060   virtual ~SwapIntervals() {}
00061 
00062   virtual bool MakeNextNeighbor(Assignment* delta, Assignment* deltadelta) {
00063     CHECK_NOTNULL(delta);
00064     while (true) {
00065       RevertChanges(true);
00066       if (!Increment()) {
00067         VLOG(1) << "finished neighborhood";
00068         return false;
00069       }
00070 
00071       std::vector<int> sequence = Sequence(current_var_);
00072       const int tmp = sequence[current_first_];
00073       sequence[current_first_] = sequence[current_second_];
00074       sequence[current_second_] = tmp;
00075       SetForwardSequence(current_var_, sequence);
00076       if (ApplyChanges(delta, deltadelta)) {
00077         VLOG(1) << "Delta = " << delta->DebugString();
00078         return true;
00079       }
00080     }
00081     return false;
00082   }
00083 
00084  protected:
00085   virtual void OnStart() {
00086     VLOG(1) << "start neighborhood";
00087     current_var_ = 0;
00088     current_first_ = 0;
00089     current_second_ = 0;
00090   }
00091 
00092  private:
00093   bool Increment() {
00094     const SequenceVar* const var = Var(current_var_);
00095     if (++current_second_ >= var->size()) {
00096       if (++current_first_ >= var->size() - 1) {
00097         current_var_++;
00098         current_first_ = 0;
00099       }
00100       current_second_ = current_first_ + 1;
00101     }
00102     return current_var_ < Size();
00103   }
00104 
00105   int current_var_;
00106   int current_first_;
00107   int current_second_;
00108 };

00191 class SequenceLns : public SequenceVarLocalSearchOperator {
00192  public:
00193   SequenceLns(const SequenceVar* const* vars,
00194               int size,
00195               int seed,
00196               int max_length)
00197       : SequenceVarLocalSearchOperator(vars, size),
00198         random_(seed),
00199         max_length_(max_length) {}
00200 
00201   virtual ~SequenceLns() {}
00202 
00203   virtual bool MakeNextNeighbor(Assignment* delta, Assignment* deltadelta) {
00204     CHECK_NOTNULL(delta);
00205     while (true) {
00206       RevertChanges(true);
00207       if (random_.Uniform(2) == 0) {
00208         FreeTimeWindow();
00209       } else {
00210         FreeTwoResources();
00211       }
00212       if (ApplyChanges(delta, deltadelta)) {
00213         VLOG(1) << "Delta = " << delta->DebugString();
00214         return true;
00215       }
00216     }
00217     return false;
00218   }
00219 
00220  private:
00221   void FreeTimeWindow() {
00222     for (int i = 0; i < Size(); ++i) {
00223       std::vector<int> sequence = Sequence(i);
00224       const int current_length =
00225           std::min(static_cast<int>(sequence.size()), max_length_);
00226       const int start_position =
00227           random_.Uniform(sequence.size() - current_length);
00228       std::vector<int> forward;
00229       for (int j = 0; j < start_position; ++j) {
00230         forward.push_back(sequence[j]);
00231       }
00232       std::vector<int> backward;
00233       for (int j = sequence.size() - 1;
00234            j >= start_position + current_length;
00235            --j) {
00236         backward.push_back(sequence[j]);
00237       }
00238       SetForwardSequence(i, forward);
00239       SetBackwardSequence(i, backward);
00240     }
00241   }
00242 
00243   void FreeTwoResources() {
00244     std::vector<int> free_sequence;
00245     SetForwardSequence(random_.Uniform(Size()), free_sequence);
00246     SetForwardSequence(random_.Uniform(Size()), free_sequence);
00247   }

00250   const int max_length_;
