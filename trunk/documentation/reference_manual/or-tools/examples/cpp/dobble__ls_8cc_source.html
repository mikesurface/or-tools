00014 // This problem is inspired by the Dobble game (aka Spot-It in the
00015 // USA).  In this game, we have 57 cards, 57 symbols, and 8 symbols
00016 // per card.  We want to assign symbols per card such that any two
00017 // cards have exactly one symbol in common. These numbers can be
00018 // generalized: we have N cards, each with K different symbols, and
00019 // there are N different symbols overall.
00020 //
00021 // This is a feasability problem. We transform that into an
00022 // optimization problem where we penalize cards whose intersection is
00023 // of cardinality different from 1. A feasible solution of the
00024 // original problem is a solution with a zero cost.
00025 //
00026 // Furthermore, we solve this problem using local search, and with a
00027 // dedicated constraint.
00028 //
00029 // The purpose of the example is to demonstrates how to write local
00030 // search operators and local search filters.

00060 // This constraint maintains:
00061 // sum_i(card1_symbol_vars[i]*card2_symbol_vars[i]) == count_var.
00062 // with all card_symbol_vars[i] being boolean variables.
00063 class SymbolsSharedByTwoCardsConstraint : public Constraint {
00064  public:
00065   // This constructor does not take any ownership on its arguments.
00066   SymbolsSharedByTwoCardsConstraint(Solver* const solver,
00067                                     const std::vector<IntVar*>& card1_symbol_vars,
00068                                     const std::vector<IntVar*>& card2_symbol_vars,
00069                                     IntVar* const num_symbols_in_common_var)
00070       : Constraint(solver),
00071         card1_symbol_vars_(card1_symbol_vars),
00072         card2_symbol_vars_(card2_symbol_vars),
00073         num_symbols_(card1_symbol_vars.size()),
00074         num_symbols_in_common_var_(num_symbols_in_common_var) {
00075     // Checks that cards have the same size.
00076     CHECK_EQ(card1_symbol_vars.size(), card2_symbol_vars.size());
00077 
00078     // Verify that we are really dealing with boolean variables.
00079     for (int i = 0; i < num_symbols_; ++i) {
00080       CHECK_GE(card1_symbol_vars_[i]->Min(), 0);
00081       CHECK_GE(card2_symbol_vars_[i]->Min(), 0);
00082       CHECK_LE(card1_symbol_vars_[i]->Max(), 1);
00083       CHECK_LE(card2_symbol_vars_[i]->Max(), 1);
00084     }
00085   }
00086 
00087   virtual ~SymbolsSharedByTwoCardsConstraint() {}
00088 
00089   // Adds observers (named Demon) to variable events. These demons are
00090   // responsible for implementing the propagation algorithm of the
00091   // constraint.
00092   virtual void Post() {
00093     // Create a demon 'global_demon' that will bind events on
00094     // variables to the calling of the 'InitialPropagate()' method. As
00095     // this method is expensive, 'global_demon' has a low priority. As
00096     // such, InitialPropagate will be called after all normal demons
00097     // and constraints have reached a fixed point. Note
00098     // that ownership of the 'global_demon' belongs to the solver.
00099     Demon* const global_demon =
00100         solver()->MakeDelayedConstraintInitialPropagateCallback(this);
00101     // Attach to all variables.
00102     for (int i = 0; i < num_symbols_; ++i) {
00103       card1_symbol_vars_[i]->WhenBound(global_demon);
00104       card2_symbol_vars_[i]->WhenBound(global_demon);
00105     }
00106     // Attach to cardinality variable.
00107     num_symbols_in_common_var_->WhenBound(global_demon);
00108   }

00110   // This is the main propagation method.
00111   //
00112   // It scans all card1_symbol_vars * card2_symbol_vars and increments 3
00113   // counters:
00114   //  - min_symbols_in_common if both booleans variables are bound to true.
00115   //  - max_symbols_in_common if both booleans are not bound to false.
00116   //
00117   // Then we know that num_symbols_in_common_var is in the range
00118   //    [min_symbols_in_common .. max_symbols_in_common].
00119   //
00120   // Now, if num_symbols_in_common_var->Max() ==
00121   // min_symbols_in_common, then all products that contribute to
00122   // max_symbols_in_common but not to min_symbols_in_common should be
00123   // set to 0.
00124   //
00125   // Conversely, if num_symbols_in_common_var->Min() ==
00126   // max_symbols_in_common, then all products that contribute to
00127   // max_symbols_in_common should be set to 1.
00128   virtual void InitialPropagate() {
00129     int max_symbols_in_common = 0;
00130     int min_symbols_in_common = 0;
00131     for (int i = 0; i < num_symbols_; ++i) {
00132       if (card1_symbol_vars_[i]->Min() == 1 &&
00133           card2_symbol_vars_[i]->Min() == 1) {
00134         min_symbols_in_common++;
00135       }
00136       if (card1_symbol_vars_[i]->Max() == 1 &&
00137           card2_symbol_vars_[i]->Max() == 1) {
00138         max_symbols_in_common++;
00139       }
00140     }
00141     num_symbols_in_common_var_->SetRange(min_symbols_in_common,
00142                                          max_symbols_in_common);
00143     // If min_symbols_in_common == max_symbols_in_common, it means
00144     // that num_symbols_in_common_var_ is already fully determined: we
00145     // have nothing to do.
00146     if (min_symbols_in_common == max_symbols_in_common) {
00147       DCHECK_EQ(min_symbols_in_common, num_symbols_in_common_var_->Max());
00148       DCHECK_EQ(min_symbols_in_common, num_symbols_in_common_var_->Min());
00149       return;
00150     }
00151     if (num_symbols_in_common_var_->Max() == min_symbols_in_common) {
00152       // All undecided product terms should be forced to 0.
00153       for (int i = 0; i < num_symbols_; ++i) {
00154         // If both Min() are 0, then we can't force either variable to
00155         // be zero (even if we know that their product is zero),
00156         // because either variable could be 1 as long as the other is
00157         // 0.
00158         if (card1_symbol_vars_[i]->Min() == 1 &&
00159             card2_symbol_vars_[i]->Min() == 0) {
00160           card2_symbol_vars_[i]->SetValue(0);
00161         } else if (card2_symbol_vars_[i]->Min() == 1 &&
00162                    card1_symbol_vars_[i]->Min() == 0) {
00163           card1_symbol_vars_[i]->SetValue(0);
00164         }
00165       }
00166     } else if (num_symbols_in_common_var_->Min() == max_symbols_in_common) {
00167       // All undecided product terms should be forced to 1.
00168       for (int i = 0; i < num_symbols_; ++i) {
00169         if (card1_symbol_vars_[i]->Max() == 1 &&
00170             card2_symbol_vars_[i]->Max() == 1) {
00171           // Note that we also force already-decided product terms,
00172           // but this doesn't change anything.
00173           card1_symbol_vars_[i]->SetValue(1);
00174           card2_symbol_vars_[i]->SetValue(1);
00175         }
00176       }
00177     }
00178   }

00180  private:
00181   std::vector<IntVar*> card1_symbol_vars_;
00182   std::vector<IntVar*> card2_symbol_vars_;
00183   const int num_symbols_;
00184   IntVar* const num_symbols_in_common_var_;
00185 };

00187 // Creates two integer variables: one that counts the number of
00188 // symbols common to two cards, and one that counts the absolute
00189 // difference between the first var and 1 (i.e. the violation of the
00190 // objective). Returns the latter (both vars are owned by the Solver
00191 // anyway).
00192 IntVar* CreateViolationVar(Solver* const solver,
00193                            const std::vector<IntVar*>& card1_symbol_vars,
00194                            const std::vector<IntVar*>& card2_symbol_vars,
00195                            int num_symbols_per_card) {

00197       solver->MakeIntVar(0, num_symbols_per_card);
00198   // RevAlloc transfers the ownership of the constraint to the solver.
00199   solver->AddConstraint(
00200       solver->RevAlloc(
00201           new SymbolsSharedByTwoCardsConstraint(solver,
00202                                                 card1_symbol_vars,
00203                                                 card2_symbol_vars,
00204                                                 num_symbols_in_common_var)));

00208 // ---------- Local Search ----------
00209 
00210 // The "local search", or "local neighborhood search", works like
00211 // this: starting from a given solution to the problem, other
00212 // solutions in its neighborhood are generated from it, some of them
00213 // might be selected (because they're better, for example) to become a
00214 // starting point for other neighborhood searches, and so on.. The
00215 // detailed search algorithm can vary and depends on the problem to
00216 // solve.
00217 //
00218 // The fundamental building block for the local search is the "search
00219 // operator", which has three fundamental methods in its API:
00220 //
00221 // - Its constructor, which keeps (mutable) references to the
00222 // solver's internal variables (here, the card symbol variables).
00223 //
00224 // - OnStart(), which is called at the start of a local search, and
00225 // after each solution (i.e. when the local search starts again from
00226 // that new solution). The solver variables are supposed to represent
00227 // a valid solution at this point. This method is used by the search
00228 // operator to initialize its state and be ready to start the
00229 // exploration of the neighborhood of the given solution.
00230 //
00231 // - MakeOneNeighbor(), which picks a neighbor of the initial
00232 // solution, and changes the solver's internal variables accordingly
00233 // to represent that new state.
00234 //
00235 // All local search operators on this problem will derive from the
00236 // parent class below, which contains some shared code to store a
00237 // compact representation of which symbols appeal on each cards.
00238 class DobbleOperator : public IntVarLocalSearchOperator {
00239  public:
00240   DobbleOperator(const IntVar* const* card_symbol_vars,
00241                  int num_vars,
00242                  int num_cards,
00243                  int num_symbols,
00244                  int num_symbols_per_card)
00245       : IntVarLocalSearchOperator(card_symbol_vars, num_vars),
00246         num_cards_(num_cards),
00247         num_symbols_(num_symbols),
00248         num_symbols_per_card_(num_symbols_per_card),
00249         symbols_per_card_(num_cards) {
00250     CHECK_GT(num_cards, 0);
00251     CHECK_GT(num_vars, 0);
00252     CHECK_GT(num_symbols, 0);
00253     CHECK_GT(num_symbols_per_card, 0);
00254     for (int card = 0; card < num_cards; ++card) {
00255       symbols_per_card_[card].assign(num_symbols_per_card, -1);
00256     }
00257   }
00258 
00259   virtual ~DobbleOperator() {}
00260 
00261  protected:
00262   // OnStart() simply stores the current symbols per card in
00263   // symbols_per_card_, and defers further initialization to the
00264   // subclass's InitNeighborhoodSearch() method.
00265   virtual void OnStart() {
00266     for (int card = 0; card < num_cards_; ++card) {
00267       int found = 0;
00268       for (int symbol = 0; symbol < num_symbols_; ++symbol) {
00269         if (Value(VarIndex(card, symbol)) == 1) {
00270           symbols_per_card_[card][found++] = symbol;
00271         }
00272       }
00273       DCHECK_EQ(num_symbols_per_card_, found);
00274     }
00275     InitNeighborhoodSearch();
00276   }

00278   virtual void InitNeighborhoodSearch() = 0;

00428 // ----- Local Search Filter -----
00429 
00430 // A filter is responsible for rejecting a local search move faster
00431 // than what the propagation of the constraint solver would do.
00432 // Its API consists in:
00433 //   - The constructor, which takes as input a reference to all the
00434 //     variables relevant to the filter.
00435 //   - OnSynchronize(), called at the beginning of the search and
00436 //     after each move to a new solution (when the local search
00437 //     restarts from it).
00438 //   - Accept(), which takes as input an attempted move (in the form
00439 //     of a Delta to tentatively apply to the variables), and returns
00440 //     true iff this move is found valid.
00441 //
00442 // To decide if a move is valid, first this DobbleFilter builds a
00443 // bitmap of symbols per card.  Then for each move, it updates the
00444 // bitmap according to the move and checks the following constraints:
00445 // - First, each card still has num_symbols_per_card symbols.  - The
00446 // cost of the assignment described by the move is better than the
00447 // current one.
00448 
00449 // After the check is done, the original bitmap is restored if the
00450 // move was rejected, so as to be ready for the next evaluation.
00451 //
00452 // Please note that this filter uses a fixed size bitset and
00453 // effectively limits the number of cards to 63, and thus the number
00454 // of symbols per card to 8.
00455 class DobbleFilter : public IntVarLocalSearchFilter {
00456  public:
00457   DobbleFilter(const IntVar* const* card_symbol_vars,
00458                int num_vars,
00459                int num_cards,
00460                int num_symbols,
00461                int num_symbols_per_card)
00462       : IntVarLocalSearchFilter(card_symbol_vars, num_vars),
00463         num_cards_(num_cards),
00464         num_symbols_(num_symbols),
00465         num_symbols_per_card_(num_symbols_per_card),
00466         temporary_bitset_(0),
00467         symbol_bitmask_per_card_(num_cards, 0),
00468         violation_costs_(num_cards_, std::vector<int>(num_cards_, 0)) {}
00469 
00470   // We build the current bitmap and the matrix of violation cost
00471   // between any two cards.
00472   virtual void OnSynchronize() {
00473     symbol_bitmask_per_card_.assign(num_cards_, 0);
00474     for (int card = 0; card < num_cards_; ++card) {
00475       for (int symbol = 0; symbol < num_symbols_; ++symbol) {
00476         if (Value(VarIndex(card, symbol))) {
00477           SetBit64(&symbol_bitmask_per_card_[card], symbol);
00478         }
00479       }
00480     }
00481     for (int card1 = 0; card1 < num_cards_; ++card1) {
00482       for (int card2 = 0; card2 < num_cards_; ++card2) {
00483         violation_costs_[card1][card2] =
00484             ViolationCost(BitCount64(symbol_bitmask_per_card_[card1] &
00485                                      symbol_bitmask_per_card_[card2]));
00486       }
00487     }
00488     DCHECK(CheckCards());
00489   }

00492   // which is the difference between the current delta and the last
00493   // delta that was given to Accept() -- but we don't use it here.
00494   virtual bool Accept(const Assignment* delta,
00495                       const Assignment* unused_deltadelta) {
00496     const Assignment::IntContainer& solution_delta = delta->IntVarContainer();
00497     const int solution_delta_size = solution_delta.Size();
00498 
00499     // The input const Assignment* delta given to Accept() may
00500     // actually contain "Deactivated" elements, which represent
00501     // variables that have been freed -- they are not bound to a
00502     // single value anymore. This happens with LNS-type (Large
00503     // Neighborhood Search) LocalSearchOperator, which are not used in
00504     // this example as of 2012-01; and we refer the reader to
00505     // ./routing.cc for an example of such LNS-type operators.
00506     //
00507     // For didactical purposes, we will assume for a moment that a
00508     // LNS-type operator might be applied. The Filter will still be
00509     // called, but our DobbleFilter here won't be able to work, since
00510     // it needs every variable to be bound (i.e. have a fixed value),
00511     // in the assignment that it considers. Therefore, we include here
00512     // a snippet of code that will detect if the input assignment is
00513     // not fully bound. For further details, read ./routing.cc -- but
00514     // we strongly advise the reader to first try and understand all
00515     // of this file.
00516     for (int i = 0; i < solution_delta_size; ++i) {
00517       if (!solution_delta.Element(i).Activated()) {
00518         VLOG(1)
00519             << "Element #" << i << " of the delta assignment given to"
00520             << " DobbleFilter::Accept() is not activated (i.e. its variable"
00521             << " is not bound to a single value anymore). This means that"
00522             << " we are in a LNS phase, and the DobbleFilter won't be able"
00523             << " to filter anything. Returning true.";
00524         return true;
00525       }
00526     }
00527     VLOG(1) << "No LNS, size = " << solution_delta_size;
00528 
00529     // Collect the set of cards that have been modified by this move.
00530     std::vector<int> touched_cards;
00531     ComputeTouchedCards(solution_delta, &touched_cards);
00532 
00533     // Check basic metrics to fail fast.
00534     if (!CheckCards()) {
00535       RestoreBitsetPerCard();
00536       DCHECK(CheckCards());
00537       VLOG(1) << "reject by size";
00538       return false;
00539     }
00540 
00541     // Compute new cost.
00542     const int cost_delta = ComputeNewCost(touched_cards);
00543 
00544     // Reset the data structure to the state before the evaluation.
00545     RestoreBitsetPerCard();
00546 
00547     // And exit (this is only valid for a greedy descent and would
00548     // reject valid moves in tabu search for instance).
00549     if (cost_delta >= 0) {
00550       VLOG(1) << "reject";
00551     }
00552     return cost_delta < 0;
00553   }

00556   // Undo information after an evaluation.
00557   struct UndoChange {
00558     UndoChange(int c, uint64 b) : card(c), bitset(b) {}
00559     int card;
00560     uint64 bitset;
00561   };

00563   int VarIndex(int card, int symbol) {
00564     return card * num_symbols_ + symbol;
00565   }

00567   void ClearBitset() {
00568     temporary_bitset_ = 0;
00569   }
00570 
00571   // For each touched card, compare against all others to compute the
00572   // delta in term of cost. We use an bitset to avoid counting twice
00573   // between two cards appearing in the local search move.
00574   int ComputeNewCost(const std::vector<int>& touched_cards) {
00575     ClearBitset();
00576     int cost_delta = 0;
00577     for (int i = 0; i < touched_cards.size(); ++i) {
00578       const int touched = touched_cards[i];
00579       SetBit64(&temporary_bitset_, touched);
00580       const uint64 card_bitset = symbol_bitmask_per_card_[touched];
00581       const std::vector<int>& row_cost = violation_costs_[touched];
00582       for (int other_card = 0; other_card < num_cards_; ++other_card) {
00583         if (!IsBitSet64(&temporary_bitset_, other_card)) {
00584           cost_delta +=
00585               ViolationCost(BitCount64(card_bitset &
00586                                        symbol_bitmask_per_card_[other_card]));
00587           cost_delta -= row_cost[other_card];
00588         }
00589       }
00590     }
00591     return cost_delta;
00592   }
00593 
00594   // Collects all card indices appearing in the local search move.
00595   void ComputeTouchedCards(const Assignment::IntContainer& solution_delta,
00596                            std::vector<int>* const touched_cards) {
00597     ClearBitset();
00598     const int solution_delta_size = solution_delta.Size();
00599     const int kUnassigned = -1;
00600     for (int index = 0; index < solution_delta_size; ++index) {
00601       int64 touched_var = kUnassigned;
00602       FindIndex(solution_delta.Element(index).Var(), &touched_var);
00603       CHECK_NE(touched_var, kUnassigned);
00604       const int card = touched_var / num_symbols_;
00605       const int symbol = touched_var % num_symbols_;
00606       const int new_value = solution_delta.Element(index).Value();
00607       if (!IsBitSet64(&temporary_bitset_, card)) {
00608         SaveRestoreInformation(card);
00609         touched_cards->push_back(card);
00610         SetBit64(&temporary_bitset_, card);
00611       }
00612       if (new_value) {
00613         SetBit64(&symbol_bitmask_per_card_[card], symbol);
00614       } else {
00615         ClearBit64(&symbol_bitmask_per_card_[card], symbol);
00616       }
00617     }
00618   }

00620   // Undo all modifications done when evaluating a move.
00621   void RestoreBitsetPerCard() {
00622     for (int i = 0; i < restore_information_.size(); ++i) {
00623       symbol_bitmask_per_card_[restore_information_[i].card] =
00624           restore_information_[i].bitset;
00625     }
00626     restore_information_.clear();
00627   }
00628 
00629   // Stores undo information for a given card.
00630   void SaveRestoreInformation(int card) {
00631     restore_information_.push_back(UndoChange(card,
00632                                               symbol_bitmask_per_card_[card]));
00633   }
00634 
00635   // Checks that after the local search move, each card would still have
00636   // num_symbols_per_card symbols on it.
00637   bool CheckCards() {
00638     for (int i = 0; i < num_cards_; ++i) {
00639       if (num_symbols_per_card_ != BitCount64(symbol_bitmask_per_card_[i])) {
00640         VLOG(1) << "card " << i << " has bitset of size "
00641                   << BitCount64(symbol_bitmask_per_card_[i]);
00642         return false;
00643       }
00644     }
00645     return true;
00646   }

00649     return (cardinality > 0 ? cardinality - 1 : 1);
00650   }

00663 void SolveDobble(int num_cards, int num_symbols, int num_symbols_per_card) {
00664   LOG(INFO) << "Solving dobble assignment problem:";
00665   LOG(INFO) << "  - " << num_cards << " cards";
00666   LOG(INFO) << "  - " << num_symbols << " symbols";
00667   LOG(INFO) << "  - " << num_symbols_per_card << " symbols per card";
00668 
00669   // Creates the solver.
00670   Solver solver("dobble");
00671   // Creates the matrix of boolean variables (cards * symbols).
00672   std::vector<std::vector<IntVar*> > card_symbol_vars(num_cards);
00673   std::vector<IntVar*> all_card_symbol_vars;
00674   for (int card_index = 0; card_index < num_cards; ++card_index) {
00675     solver.MakeBoolVarArray(num_symbols,
00676                             StringPrintf("card_%i_", card_index),
00677                             &card_symbol_vars[card_index]);
00678     for (int symbol_index = 0;
00679          symbol_index < num_symbols;
00680          ++symbol_index) {
00681       all_card_symbol_vars.push_back(
00682           card_symbol_vars[card_index][symbol_index]);
00683     }
00684   }
00685   // Creates cardinality intersection variables and remember the
00686   // violation variables.
00687   std::vector<IntVar*> violation_vars;
00688   for (int card1 = 0; card1 < num_cards; ++card1) {
00689     for (int card2 = 0; card2 < num_cards; ++card2) {
00690       if (card1 != card2) {
00691         violation_vars.push_back(CreateViolationVar(&solver,
00692                                                     card_symbol_vars[card1],
00693                                                     card_symbol_vars[card2],
00694                                                     num_symbols_per_card));
00695       }
00696     }
00697   }
00698   // Create the objective variable.
00699   IntVar* const objective_var = solver.MakeSum(violation_vars)->Var();
00700 
00701   // Add constraint: there must be exactly num_symbols_per_card
00702   // symbols per card.
00703   for (int card = 0; card < num_cards; ++card) {
00704     solver.AddConstraint(solver.MakeSumEquality(card_symbol_vars[card],
00705                                                 num_symbols_per_card));
00706   }
00707 
00708   // IMPORTANT OPTIMIZATION:
00709   // Add constraint: each symbol appears on exactly
00710   // num_symbols_per_card cards (i.e. symbols are evenly
00711   // distributed). This constraint is actually redundant, because it
00712   // is a (non-trivial) consequence of the other constraints and of
00713   // the model. But adding it makes the search go faster.
00714   for (int symbol_index = 0; symbol_index < num_symbols; ++symbol_index) {
00715     std::vector<IntVar*> tmp;
00716     for (int card_index = 0; card_index < num_cards; ++card_index) {
00717       tmp.push_back(card_symbol_vars[card_index][symbol_index]);
00718      }
00719     solver.AddConstraint(solver.MakeSumEquality(tmp, num_symbols_per_card));
00720   }
00721 
00722   // Search.
00723   LOG(INFO) << "Solving with Local Search";
00724   LOG(INFO) << "  - time limit = " << FLAGS_time_limit_in_ms << " ms";
00725 
00726   // Start a DecisionBuilder phase to find a first solution, using the
00727   // strategy "Pick some random, yet unassigned card symbol variable
00728   // and set its value to 1".
00729   DecisionBuilder* const build_db = solver.MakePhase(
00730       all_card_symbol_vars,
00731       Solver::CHOOSE_RANDOM,      // Solver::IntVarStrategy
00732       Solver::ASSIGN_MAX_VALUE);  // Solver::IntValueStrategy
00733 
00734   // Creates local search operators.
00735   std::vector<LocalSearchOperator*> operators;
00736   LocalSearchOperator* const switch_operator =
00737       solver.RevAlloc(new SwapSymbols(all_card_symbol_vars.data(),
00738                                       all_card_symbol_vars.size(),
00739                                       num_cards,
00740                                       num_symbols,
00741                                       num_symbols_per_card));
00742   operators.push_back(switch_operator);
00743   LOG(INFO) << "  - add switch operator";
00744   if (FLAGS_num_swaps > 0) {
00745     LocalSearchOperator* const swaps_operator =
00746         solver.RevAlloc(new SwapSymbolsOnCardPairs(all_card_symbol_vars.data(),
00747                                                    all_card_symbol_vars.size(),
00748                                                    num_cards,
00749                                                    num_symbols,
00750                                                    num_symbols_per_card,
00751                                                    FLAGS_num_swaps));
00752     operators.push_back(swaps_operator);
00753     LOG(INFO) << "  - add swaps operator with at most "
00754               << FLAGS_num_swaps << " swaps";
00755   }
00756 
00757   // Creates filter.
00758   std::vector<LocalSearchFilter*> filters;
00759   if (FLAGS_use_filter) {
00760     filters.push_back(solver.RevAlloc(
00761         new DobbleFilter(all_card_symbol_vars.data(),
00762                          all_card_symbol_vars.size(),
00763                          num_cards,
00764                          num_symbols,
00765                          num_symbols_per_card)));
00766   }
00767 
00768   // Main decision builder that regroups the first solution decision
00769   // builder and the combination of local search operators and
00770   // filters.
00771   DecisionBuilder* const final_db =
00772       solver.MakeLocalSearchPhase(
00773           all_card_symbol_vars,
00774           build_db,
00775           solver.MakeLocalSearchPhaseParameters(
00776               solver.ConcatenateOperators(operators, true),
00777               NULL,  // Sub decision builder, not needed here.
00778               NULL,  // Limit the search for improving move, we will stop
00779                      // the exploration of the local search at the first
00780                      // improving solution (first accept).
00781               filters));
00782 
00783 
00784   std::vector<SearchMonitor*> monitors;
00785   // Optimize var search monitor.
00786   OptimizeVar* const optimize = solver.MakeMinimize(objective_var, 1);
00787   monitors.push_back(optimize);
00788 
00789   // Search log.
00790   SearchMonitor* const log = solver.MakeSearchLog(100000, optimize);
00791   monitors.push_back(log);
00792 
00793   // Search limit.
00794   SearchLimit* const time_limit =
00795       solver.MakeLimit(FLAGS_time_limit_in_ms, kint64max, kint64max, kint64max);
00796   monitors.push_back(time_limit);
00797 
00798   // And solve!
00799   solver.Solve(final_db, monitors);
00800 }

00804   google::ParseCommandLineFlags(&argc, &argv, true);
00805   // These constants comes directly from the dobble game.
00806   // There are actually 55 cards, but we can create up to 57 cards.

00813   return 0;
