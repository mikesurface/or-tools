00014 // This model implements a simple jobshop problem.
00015 //
00016 // A jobshop is a standard scheduling problem where you must schedule a
00017 // set of jobs on a set of machines.  Each job is a sequence of tasks
00018 // (a task can only start when the preceding task finished), each of
00019 // which occupies a single specific machine during a specific
00020 // duration. Therefore, a job is simply given by a sequence of pairs
00021 // (machine id, duration).
00022 
00023 // The objective is to minimize the 'makespan', which is the duration
00024 // between the start of the first task (across all machines) and the
00025 // completion of the last task (across all machines).
00026 //
00027 // This will be modelled by sets of intervals variables (see class
00028 // IntervalVar in constraint_solver/constraint_solver.h), one per
00029 // task, representing the [start_time, end_time] of the task.  Tasks
00030 // in the same job will be linked by precedence constraints.  Tasks on
00031 // the same machine will be covered by Sequence constraints.
00032 //
00033 // Search will then be applied on the sequence constraints.
00034

00059 namespace operations_research {
00060 void Jobshop(const JobShopData& data) {
00061   Solver solver("jobshop");
00062   const int machine_count = data.machine_count();
00063   const int job_count = data.job_count();
00064   const int horizon = data.horizon();
00065 
00066   // ----- Creates all Intervals and vars -----
00067 
00068   // Stores all tasks attached interval variables per job.
00069   std::vector<std::vector<IntervalVar*> > jobs_to_tasks(job_count);
00070   // machines_to_tasks stores the same interval variables as above, but
00071   // grouped my machines instead of grouped by jobs.
00072   std::vector<std::vector<IntervalVar*> > machines_to_tasks(machine_count);
00073 
00074   // Creates all individual interval variables.
00075   for (int job_id = 0; job_id < job_count; ++job_id) {
00076     const std::vector<JobShopData::Task>& tasks = data.TasksOfJob(job_id);
00077     for (int task_index = 0; task_index < tasks.size(); ++task_index) {
00078       const JobShopData::Task& task = tasks[task_index];
00079       CHECK_EQ(job_id, task.job_id);
00080       const string name = StringPrintf("J%dM%dI%dD%d",
00081                                        task.job_id,
00082                                        task.machine_id,
00083                                        task_index,
00084                                        task.duration);
00085       IntervalVar* const one_task =
00086           solver.MakeFixedDurationIntervalVar(0,
00087                                               horizon,
00088                                               task.duration,
00089                                               false,
00090                                               name);
00091       jobs_to_tasks[task.job_id].push_back(one_task);
00092       machines_to_tasks[task.machine_id].push_back(one_task);
00093     }
00094   }
00095 
00096   // ----- Creates model -----
00097 
00098   // Creates precedences inside jobs.
00099   for (int job_id = 0; job_id < job_count; ++job_id) {
00100     const int task_count = jobs_to_tasks[job_id].size();
00101     for (int task_index = 0; task_index < task_count - 1; ++task_index) {
00102       IntervalVar* const t1 = jobs_to_tasks[job_id][task_index];
00103       IntervalVar* const t2 = jobs_to_tasks[job_id][task_index + 1];
00104       Constraint* const prec =
00105           solver.MakeIntervalVarRelation(t2, Solver::STARTS_AFTER_END, t1);
00106       solver.AddConstraint(prec);
00107     }
00108   }
00109 
00110   // Adds disjunctive constraints on unary resources, and creates
00111   // sequence variables. A sequence variable is a dedicated variable
00112   // whose job is to sequence interval variables.
00113   std::vector<SequenceVar*> all_sequences;
00114   for (int machine_id = 0; machine_id < machine_count; ++machine_id) {
00115     const string name = StringPrintf("Machine_%d", machine_id);
00116     DisjunctiveConstraint* const ct =
00117         solver.MakeDisjunctiveConstraint(machines_to_tasks[machine_id], name);
00118     solver.AddConstraint(ct);
00119     all_sequences.push_back(ct->MakeSequenceVar());
00120   }
00121 
00122   // Creates array of end_times of jobs.
00123   std::vector<IntVar*> all_ends;
00124   for (int job_id = 0; job_id < job_count; ++job_id) {
00125     const int task_count = jobs_to_tasks[job_id].size();
00126     IntervalVar* const task = jobs_to_tasks[job_id][task_count - 1];
00127     all_ends.push_back(task->EndExpr()->Var());
00128   }
00129 
00130   // Objective: minimize the makespan (maximum end times of all tasks)
00131   // of the problem.
00132   IntVar* const objective_var = solver.MakeMax(all_ends)->Var();
00133   OptimizeVar* const objective_monitor = solver.MakeMinimize(objective_var, 1);
00134 
00135   // ----- Search monitors and decision builder -----
00136 
00137   // This decision builder will rank all tasks on all machines.
00138   DecisionBuilder* const sequence_phase =
00139       solver.MakePhase(all_sequences, Solver::SEQUENCE_DEFAULT);
00140 
00141   // After the ranking of tasks, the schedule is still loose and any
00142   // task can be postponed at will. But, because the problem is now a PERT
00143   // (http://en.wikipedia.org/wiki/Program_Evaluation_and_Review_Technique),
00144   // we can schedule each task at its earliest start time. This is
00145   // conveniently done by fixing the objective variable to its
00146   // minimum value.
00147   DecisionBuilder* const obj_phase =
00148       solver.MakePhase(objective_var,
00149                        Solver::CHOOSE_FIRST_UNBOUND,
00150                        Solver::ASSIGN_MIN_VALUE);
00151 
00152   // The main decision builder (ranks all tasks, then fixes the
00153   // objective_variable).
00154   DecisionBuilder* const main_phase =
00155       solver.Compose(sequence_phase, obj_phase);
00156 
00157   // Search log.
00158   const int kLogFrequency = 1000000;
00159   SearchMonitor* const search_log =
00160       solver.MakeSearchLog(kLogFrequency, objective_monitor);
00161 
00162   SearchLimit* limit = NULL;
00163   if (FLAGS_time_limit_in_ms > 0) {
00164     limit = solver.MakeTimeLimit(FLAGS_time_limit_in_ms);
00165   }
00166 
00167   SolutionCollector* const collector =
00168       solver.MakeLastSolutionCollector();
00169   collector->Add(all_sequences);
00170 
00171   // Search.
00172   if (solver.Solve(main_phase,
00173                    search_log,
00174                    objective_monitor,
00175                    limit,
00176                    collector)) {
00177     for (int m = 0; m < machine_count; ++m) {
00178       SequenceVar* const seq = all_sequences[m];
00179       LOG(INFO) << seq->name() << ": "
00180                 << IntVectorToString(collector->ForwardSequence(0, seq), ", ");
00181     }
00182   }
00183 }
00184 }  // namespace operations_research
