00014 // This model implements a simple jobshop problem.
00015 //
00016 // A jobshop is a standard scheduling problem where you must schedule a
00017 // set of jobs on a set of machines.  Each job is a sequence of tasks
00018 // (a task can only start when the preceding task finished), each of
00019 // which occupies a single specific machine during a specific
00020 // duration. Therefore, a job is simply given by a sequence of pairs
00021 // (machine id, duration).
00022 
00023 // The objective is to minimize the 'makespan', which is the duration
00024 // between the start of the first task (across all machines) and the
00025 // completion of the last task (across all machines).
00026 //
00027 // This will be modelled by sets of intervals variables (see class
00028 // IntervalVar in constraint_solver/constraint_solver.h), one per
00029 // task, representing the [start_time, end_time] of the task.  Tasks
00030 // in the same job will be linked by precedence constraints.  Tasks on
00031 // the same machine will be covered by Sequence constraints.
00032 //
00033 // Search will be implemented as local search on the sequence variables.

00071 void JobshopLs(const JobShopData& data) {
00072   Solver solver("jobshop");
00073   const int machine_count = data.machine_count();
00074   const int job_count = data.job_count();
00075   const int horizon = data.horizon();
00076 
00077   // ----- Creates all Intervals and vars -----
00078 
00079   // Stores all tasks attached interval variables per job.
00080   std::vector<std::vector<IntervalVar*> > jobs_to_tasks(job_count);
00081   // machines_to_tasks stores the same interval variables as above, but
00082   // grouped my machines instead of grouped by jobs.
00083   std::vector<std::vector<IntervalVar*> > machines_to_tasks(machine_count);
00084 
00085   // Creates all individual interval variables.
00086   for (int job_id = 0; job_id < job_count; ++job_id) {
00087     const std::vector<JobShopData::Task>& tasks = data.TasksOfJob(job_id);
00088     for (int task_index = 0; task_index < tasks.size(); ++task_index) {
00089       const JobShopData::Task& task = tasks[task_index];
00090       CHECK_EQ(job_id, task.job_id);
00091       const string name = StringPrintf("J%dM%dI%dD%d",
00092                                        task.job_id,
00093                                        task.machine_id,
00094                                        task_index,
00095                                        task.duration);
00096       IntervalVar* const one_task =
00097           solver.MakeFixedDurationIntervalVar(0,
00098                                               horizon,
00099                                               task.duration,
00100                                               false,
00101                                               name);
00102       jobs_to_tasks[task.job_id].push_back(one_task);
00103       machines_to_tasks[task.machine_id].push_back(one_task);
00104     }
00105   }
00106 
00107   // ----- Creates model -----
00108 
00109   // Creates precedences inside jobs.
00110   for (int job_id = 0; job_id < job_count; ++job_id) {
00111     const int task_count = jobs_to_tasks[job_id].size();
00112     for (int task_index = 0; task_index < task_count - 1; ++task_index) {
00113       IntervalVar* const t1 = jobs_to_tasks[job_id][task_index];
00114       IntervalVar* const t2 = jobs_to_tasks[job_id][task_index + 1];
00115       Constraint* const prec =
00116           solver.MakeIntervalVarRelation(t2, Solver::STARTS_AFTER_END, t1);
00117       solver.AddConstraint(prec);
00118     }
00119   }
00120 
00121   // Adds disjunctive constraints on unary resources, and creates
00122   // sequence variables. A sequence variable is a dedicated variable
00123   // whose job is to sequence interval variables.
00124   std::vector<SequenceVar*> all_sequences;
00125   for (int machine_id = 0; machine_id < machine_count; ++machine_id) {
00126     const string name = StringPrintf("Machine_%d", machine_id);
00127     DisjunctiveConstraint* const ct =
00128         solver.MakeDisjunctiveConstraint(machines_to_tasks[machine_id], name);
00129     solver.AddConstraint(ct);
00130     all_sequences.push_back(ct->MakeSequenceVar());
00131   }
00132 
00133   // Creates array of end_times of jobs.
00134   std::vector<IntVar*> all_ends;
00135   for (int job_id = 0; job_id < job_count; ++job_id) {
00136     const int task_count = jobs_to_tasks[job_id].size();
00137     IntervalVar* const task = jobs_to_tasks[job_id][task_count - 1];
00138     all_ends.push_back(task->EndExpr()->Var());
00139   }
00140 
00141   // Objective: minimize the makespan (maximum end times of all tasks)
00142   // of the problem.
00143   IntVar* const objective_var = solver.MakeMax(all_ends)->Var();
00144 
00145   // ----- Search monitors and decision builder -----
00146 
00147   // This decision builder will rank all tasks on all machines.
00148   DecisionBuilder* const sequence_phase =
00149       solver.MakePhase(all_sequences, Solver::SEQUENCE_DEFAULT);
00150 
00151   // After the ranking of tasks, the schedule is still loose and any
00152   // task can be postponed at will. But, because the problem is now a PERT
00153   // (http://en.wikipedia.org/wiki/Program_Evaluation_and_Review_Technique),
00154   // we can schedule each task at its earliest start time. This is
00155   // conveniently done by fixing the objective variable to its
00156   // minimum value.
00157   DecisionBuilder* const obj_phase =
00158       solver.MakePhase(objective_var,
00159                        Solver::CHOOSE_FIRST_UNBOUND,
00160                        Solver::ASSIGN_MIN_VALUE);
00161 
00162   Assignment* const first_solution = solver.MakeAssignment();
00163   first_solution->Add(all_sequences);
00164   first_solution->AddObjective(objective_var);
00165   // Store the first solution in the 'solution' object.
00166   DecisionBuilder* const store_db = solver.MakeStoreAssignment(first_solution);
00167 
00168   // The main decision builder (ranks all tasks, then fixes the
00169   // objective_variable).
00170   DecisionBuilder* const first_solution_phase =
00171       solver.Compose(sequence_phase, obj_phase, store_db);
00172 
00173   LOG(INFO) << "Looking for the first solution";
00174   const bool first_solution_found = solver.Solve(first_solution_phase);
00175   if (first_solution_found) {
00176     LOG(INFO) << "Solution found with makespan = "
00177               << first_solution->ObjectiveValue();
00178   } else {
00179     LOG(INFO) << "No initial solution found!";
00180     return;
00181   }
00182 
00183   LOG(INFO) << "Switching to local search";
00184   std::vector<LocalSearchOperator*> operators;
00185   LOG(INFO) << "  - use swap operator";
00186   LocalSearchOperator* const swap_operator =
00187       solver.RevAlloc(new SwapIntervals(all_sequences.data(),
00188                                         all_sequences.size()));
00189   operators.push_back(swap_operator);
00190   LOG(INFO) << "  - use shuffle operator with a max length of "
00191             << FLAGS_shuffle_length;
00192   LocalSearchOperator* const shuffle_operator =
00193       solver.RevAlloc(new ShuffleIntervals(all_sequences.data(),
00194                                            all_sequences.size(),
00195                                            FLAGS_shuffle_length));
00196   operators.push_back(shuffle_operator);
00197   LOG(INFO) << "  - use free sub sequences of length "
00198             << FLAGS_sub_sequence_length << " lns operator";
00199   LocalSearchOperator* const lns_operator =
00200       solver.RevAlloc(new SequenceLns(all_sequences.data(),
00201                                       all_sequences.size(),
00202                                       FLAGS_lns_seed,
00203                                       FLAGS_sub_sequence_length));
00204   operators.push_back(lns_operator);
00205 
00206   // Creates the local search decision builder.
00207   LocalSearchOperator* const concat =
00208       solver.ConcatenateOperators(operators, true);
00209 
00210   SearchLimit* const ls_limit =
00211       solver.MakeLimit(kint64max, FLAGS_lns_limit, kint64max, kint64max);
00212   DecisionBuilder* const random_sequence_phase =
00213       solver.MakePhase(all_sequences, Solver::CHOOSE_RANDOM_RANK_FORWARD);
00214   DecisionBuilder* const ls_db =
00215       solver.MakeSolveOnce(solver.Compose(random_sequence_phase, obj_phase),
00216                            ls_limit);
00217 
00218   LocalSearchPhaseParameters* const parameters =
00219       solver.MakeLocalSearchPhaseParameters(concat, ls_db);
00220   DecisionBuilder* const final_db =
00221       solver.MakeLocalSearchPhase(first_solution, parameters);
00222 
00223   OptimizeVar* const objective_monitor = solver.MakeMinimize(objective_var, 1);
00224 
00225   // Search log.
00226   const int kLogFrequency = 1000000;
00227   SearchMonitor* const search_log =
00228       solver.MakeSearchLog(kLogFrequency, objective_monitor);
00229 
00230   SearchLimit* const limit = FLAGS_time_limit_in_ms > 0 ?
00231       solver.MakeTimeLimit(FLAGS_time_limit_in_ms) :
00232       NULL;
00233 
00234   // Search.
00235   solver.Solve(final_db, search_log, objective_monitor, limit);
00236 }
