00014 // This model implements a simple jobshop problem with
00015 // earliness-tardiness costs.
00016 //
00017 // A earliness-tardiness jobshop is a standard scheduling problem where
00018 // you must schedule a set of jobs on a set of machines.  Each job is
00019 // a sequence of tasks (a task can only start when the preceding task
00020 // finished), each of which occupies a single specific machine during
00021 // a specific duration. Therefore, a job is a sequence of pairs
00022 // (machine id, duration), along with a release data (minimum start
00023 // date of the first task of the job, and due data (end time of the
00024 // last job) with a tardiness linear penalty.
00025 
00026 // The objective is to minimize the sum of early-tardy penalties for each job.
00027 //
00028 // This will be modelled by sets of intervals variables (see class
00029 // IntervalVar in constraint_solver/constraint_solver.h), one per
00030 // task, representing the [start_time, end_time] of the task.  Tasks
00031 // in the same job will be linked by precedence constraints.  Tasks on
00032 // the same machine will be covered by Sequence constraints.

00083                 const std::vector<SequenceVar*>& all_sequences,

00092   virtual Decision* Next(Solver* const solver) {
00093     mp_solver_.Clear();
00094     std::vector<std::vector<MPVariable*> > all_vars;
00095     hash_map<IntervalVar*, MPVariable*> mapping;
00096     const double infinity = mp_solver_.infinity();
00097     all_vars.resize(all_sequences_.size());
00098 
00099     // Creates the MP Variables.
00100     for (int s = 0; s < jobs_to_tasks_.size(); ++s) {
00101       for (int t = 0; t < jobs_to_tasks_[s].size(); ++t) {
00102         IntervalVar* const task = jobs_to_tasks_[s][t];
00103         const string name = StringPrintf("J%dT%d", s, t);
00104         MPVariable* const var =
00105             mp_solver_.MakeIntVar(task->StartMin(), task->StartMax(), name);
00106         mapping[task] = var;
00107       }
00108     }
00109 
00110       // Adds the jobs precedence constraints.
00111     for (int j = 0; j < jobs_to_tasks_.size(); ++j) {
00112       for (int t = 0; t < jobs_to_tasks_[j].size() - 1; ++t) {
00113         IntervalVar* const first_task = jobs_to_tasks_[j][t];
00114         const int duration = first_task->DurationMax();
00115         IntervalVar* const second_task = jobs_to_tasks_[j][t + 1];
00116         MPVariable* const first_var = mapping[first_task];
00117         MPVariable* const second_var = mapping[second_task];
00118         MPConstraint* const ct =
00119             mp_solver_.MakeRowConstraint(duration, infinity);
00120         ct->SetCoefficient(second_var, 1.0);
00121         ct->SetCoefficient(first_var, -1.0);
00122       }
00123     }
00124 
00125     // Adds the ranked machines constraints.
00126     for (int s = 0; s < all_sequences_.size(); ++s) {
00127       SequenceVar* const sequence = all_sequences_[s];
00128       std::vector<int> rank_firsts;
00129       std::vector<int> rank_lasts;
00130       std::vector<int> unperformed;
00131       sequence->FillSequence(&rank_firsts, &rank_lasts, &unperformed);
00132       CHECK_EQ(0, rank_lasts.size());
00133       CHECK_EQ(0, unperformed.size());
00134       for (int i = 0; i < rank_firsts.size() - 1; ++i) {
00135         IntervalVar* const first_task = sequence->Interval(rank_firsts[i]);
00136         const int duration = first_task->DurationMax();
00137         IntervalVar* const second_task = sequence->Interval(rank_firsts[i + 1]);
00138         MPVariable* const first_var = mapping[first_task];
00139         MPVariable* const second_var = mapping[second_task];
00140         MPConstraint* const ct =
00141             mp_solver_.MakeRowConstraint(duration, infinity);
00142         ct->SetCoefficient(second_var, 1.0);
00143         ct->SetCoefficient(first_var, -1.0);
00144       }
00145     }
00146 
00147     // Creates penalty terms and objective.
00148     std::vector<MPVariable*> terms;
00149     mp_solver_.MakeIntVarArray(jobs_to_tasks_.size(),
00150                                0,
00151                                infinity,
00152                                "terms",
00153                                &terms);
00154     for (int j = 0; j < jobs_to_tasks_.size(); ++j) {
00155       mp_solver_.MutableObjective()->SetCoefficient(terms[j], 1.0);
00156     }
00157     mp_solver_.MutableObjective()->SetMinimization();
00158 
00159     // Forces penalty terms to be above late and early costs.
00160     for (int j = 0; j < jobs_to_tasks_.size(); ++j) {
00161       IntervalVar* const last_task = jobs_to_tasks_[j].back();
00162       const int duration = last_task->DurationMin();
00163       MPVariable* const mp_start = mapping[last_task];
00164       const Job& job = data_.GetJob(j);
00165       const int ideal_start = job.due_date - duration;
00166       const int early_offset = job.early_cost * ideal_start;
00167       MPConstraint* const early_ct =
00168           mp_solver_.MakeRowConstraint(early_offset, infinity);
00169       early_ct->SetCoefficient(terms[j], 1);
00170       early_ct->SetCoefficient(mp_start, job.early_cost);
00171 
00172       const int tardy_offset = job.tardy_cost * ideal_start;
00173       MPConstraint* const tardy_ct =
00174           mp_solver_.MakeRowConstraint(-tardy_offset, infinity);
00175       tardy_ct->SetCoefficient(terms[j], 1);
00176       tardy_ct->SetCoefficient(mp_start, -job.tardy_cost);
00177     }
00178 
00179     // Solve.
00180     CHECK_EQ(MPSolver::OPTIMAL, mp_solver_.Solve());
00181 
00182     // Inject MIP solution into the CP part.
00183     VLOG(1) << "MP cost = " << mp_solver_.objective_value();
00184     for (int j = 0; j < jobs_to_tasks_.size(); ++j) {
00185       for (int t = 0; t < jobs_to_tasks_[j].size(); ++t) {
00186         IntervalVar* const first_task = jobs_to_tasks_[j][t];
00187         MPVariable* const first_var = mapping[first_task];
00188         const int date = first_var->solution_value();
00189         first_task->SetStartRange(date, date);
00190       }
00191     }
00192     return NULL;
00193   }

00206 void EtJobShop(const EtJobShopData& data) {
00207   Solver solver("et_jobshop");
00208   const int machine_count = data.machine_count();
00209   const int job_count = data.job_count();
00210   const int horizon = data.horizon();
00211 
00212   // ----- Creates all Intervals and vars -----
00213 
00214   // Stores all tasks attached interval variables per job.
00215   std::vector<std::vector<IntervalVar*> > jobs_to_tasks(job_count);
00216   // machines_to_tasks stores the same interval variables as above, but
00217   // grouped my machines instead of grouped by jobs.
00218   std::vector<std::vector<IntervalVar*> > machines_to_tasks(machine_count);
00219 
00220   // Creates all individual interval variables.
00221   for (int job_id = 0; job_id < job_count; ++job_id) {
00222     const Job& job = data.GetJob(job_id);
00223     const std::vector<Task>& tasks = job.all_tasks;
00224     for (int task_index = 0; task_index < tasks.size(); ++task_index) {
00225       const Task& task = tasks[task_index];
00226       CHECK_EQ(job_id, task.job_id);
00227       const string name = StringPrintf("J%dM%dI%dD%d",
00228                                        task.job_id,
00229                                        task.machine_id,
00230                                        task_index,
00231                                        task.duration);
00232       IntervalVar* const one_task =
00233           solver.MakeFixedDurationIntervalVar(0,
00234                                               horizon,
00235                                               task.duration,
00236                                               false,
00237                                               name);
00238       jobs_to_tasks[task.job_id].push_back(one_task);
00239       machines_to_tasks[task.machine_id].push_back(one_task);
00240     }
00241   }
00242 
00243   // ----- Creates model -----
00244 
00245   // Creates precedences inside jobs.
00246   for (int job_id = 0; job_id < job_count; ++job_id) {
00247     const int task_count = jobs_to_tasks[job_id].size();
00248     for (int task_index = 0; task_index < task_count - 1; ++task_index) {
00249       IntervalVar* const t1 = jobs_to_tasks[job_id][task_index];
00250       IntervalVar* const t2 = jobs_to_tasks[job_id][task_index + 1];
00251       Constraint* const prec =
00252           solver.MakeIntervalVarRelation(t2, Solver::STARTS_AFTER_END, t1);
00253       solver.AddConstraint(prec);
00254     }
00255   }
00256 
00257   // Add release date.
00258   for (int job_id = 0; job_id < job_count; ++job_id) {
00259     const Job& job = data.GetJob(job_id);
00260     IntervalVar* const t = jobs_to_tasks[job_id][0];
00261     Constraint* const prec =
00262         solver.MakeIntervalVarRelation(t,
00263                                        Solver::STARTS_AFTER,
00264                                        job.release_date);
00265     solver.AddConstraint(prec);
00266   }
00267 
00268   std::vector<IntVar*> penalties;
00269   for (int job_id = 0; job_id < job_count; ++job_id) {
00270     const Job& job = data.GetJob(job_id);
00271     IntervalVar* const t = jobs_to_tasks[job_id][machine_count - 1];
00272     IntVar* const penalty =
00273         solver.MakeConvexPiecewiseExpr(t->EndExpr(),
00274                                        job.early_cost,
00275                                        job.due_date,
00276                                        job.due_date,
00277                                        job.tardy_cost)->Var();
00278     penalties.push_back(penalty);
00279   }
00280 
00281   // Adds disjunctive constraints on unary resources, and creates
00282   // sequence variables. A sequence variable is a dedicated variable
00283   // whose job is to sequence interval variables.
00284   std::vector<SequenceVar*> all_sequences;
00285   for (int machine_id = 0; machine_id < machine_count; ++machine_id) {
00286     const string name = StringPrintf("Machine_%d", machine_id);
00287     DisjunctiveConstraint* const ct =
00288         solver.MakeDisjunctiveConstraint(machines_to_tasks[machine_id], name);
00289     solver.AddConstraint(ct);
00290     all_sequences.push_back(ct->MakeSequenceVar());
00291   }
00292 
00293   // Objective: minimize the weighted penalties.
00294   IntVar* const objective_var = solver.MakeSum(penalties)->Var();
00295   OptimizeVar* const objective_monitor = solver.MakeMinimize(objective_var, 1);
00296 
00297   // ----- Search monitors and decision builder -----
00298 
00299   // This decision builder will rank all tasks on all machines.
00300   DecisionBuilder* const sequence_phase =
00301       solver.MakePhase(all_sequences, Solver::CHOOSE_MIN_SLACK_RANK_FORWARD);
00302 
00303   // After the ranking of tasks, the schedule is still loose and any
00304   // task can be postponed at will. But, because the problem is now a PERT
00305   // (http://en.wikipedia.org/wiki/Program_Evaluation_and_Review_Technique),
00306   // we can schedule each task at its earliest start time. This is
00307   // conveniently done by fixing the objective variable to its
00308   // minimum value.
00309   DecisionBuilder* const obj_phase =
00310       FLAGS_time_placement ?
00311       solver.RevAlloc(new TimePlacement(data, all_sequences, jobs_to_tasks)) :
00312       solver.MakePhase(objective_var,
00313                        Solver::CHOOSE_FIRST_UNBOUND,
00314                        Solver::ASSIGN_MIN_VALUE);
00315 
00316   if (FLAGS_use_ls) {
00317     Assignment* const first_solution = solver.MakeAssignment();
00318     first_solution->Add(all_sequences);
00319     first_solution->AddObjective(objective_var);
00320     // Store the first solution in the 'solution' object.
00321     DecisionBuilder* const store_db =
00322         solver.MakeStoreAssignment(first_solution);
00323 
00324     // The main decision builder (ranks all tasks, then fixes the
00325     // objective_variable).
00326     DecisionBuilder* const first_solution_phase =
00327         solver.Compose(sequence_phase, obj_phase, store_db);
00328 
00329     LOG(INFO) << "Looking for the first solution";
00330     const bool first_solution_found = solver.Solve(first_solution_phase);
00331     if (first_solution_found) {
00332       LOG(INFO) << "Solution found with penalty cost of = "
00333                 << first_solution->ObjectiveValue();
00334     } else {
00335       LOG(INFO) << "No initial solution found!";
00336       return;
00337     }
00338 
00339     LOG(INFO) << "Switching to local search";
00340     std::vector<LocalSearchOperator*> operators;
00341     LOG(INFO) << "  - use swap operator";
00342     LocalSearchOperator* const swap_operator =
00343         solver.RevAlloc(new SwapIntervals(all_sequences.data(),
00344                                           all_sequences.size()));
00345     operators.push_back(swap_operator);
00346     LOG(INFO) << "  - use shuffle operator with a max length of "
00347               << FLAGS_shuffle_length;
00348     LocalSearchOperator* const shuffle_operator =
00349         solver.RevAlloc(new ShuffleIntervals(all_sequences.data(),
00350                                              all_sequences.size(),
00351                                              FLAGS_shuffle_length));
00352     operators.push_back(shuffle_operator);
00353     LOG(INFO) << "  - use free sub sequences of length "
00354               << FLAGS_sub_sequence_length << " lns operator";
00355     LocalSearchOperator* const lns_operator =
00356         solver.RevAlloc(new SequenceLns(all_sequences.data(),
00357                                         all_sequences.size(),
00358                                         FLAGS_lns_seed,
00359                                         FLAGS_sub_sequence_length));
00360     operators.push_back(lns_operator);
00361 
00362     // Creates the local search decision builder.
00363     LocalSearchOperator* const concat =
00364         solver.ConcatenateOperators(operators, true);
00365 
00366     SearchLimit* const ls_limit =
00367         solver.MakeLimit(kint64max, FLAGS_lns_limit, kint64max, kint64max);
00368     DecisionBuilder* const random_sequence_phase =
00369         solver.MakePhase(all_sequences, Solver::CHOOSE_RANDOM_RANK_FORWARD);
00370     DecisionBuilder* const ls_db =
00371         solver.MakeSolveOnce(solver.Compose(random_sequence_phase, obj_phase),
00372                              ls_limit);
00373 
00374     LocalSearchPhaseParameters* const parameters =
00375         solver.MakeLocalSearchPhaseParameters(concat, ls_db);
00376     DecisionBuilder* const final_db =
00377         solver.MakeLocalSearchPhase(first_solution, parameters);
00378 
00379     OptimizeVar* const objective_monitor =
00380         solver.MakeMinimize(objective_var, 1);
00381 
00382     // Search log.
00383     const int kLogFrequency = 1000000;
00384     SearchMonitor* const search_log =
00385         solver.MakeSearchLog(kLogFrequency, objective_monitor);
00386 
00387     SearchLimit* const limit = FLAGS_time_limit_in_ms > 0 ?
00388         solver.MakeTimeLimit(FLAGS_time_limit_in_ms) :
00389         NULL;
00390 
00391     // Search.
00392     solver.Solve(final_db, search_log, objective_monitor, limit);
00393   } else {
00394     // The main decision builder (ranks all tasks, then fixes the
00395     // objective_variable).
00396     DecisionBuilder* const main_phase =
00397         solver.Compose(sequence_phase, obj_phase);
00398 
00399     // Search log.
00400     const int kLogFrequency = 1000000;
00401     SearchMonitor* const search_log =
00402         solver.MakeSearchLog(kLogFrequency, objective_monitor);
00403 
00404     SearchLimit* limit = NULL;
00405     if (FLAGS_time_limit_in_ms > 0) {
00406       limit = solver.MakeTimeLimit(FLAGS_time_limit_in_ms);
00407     }
00408 
00409     // Search.
00410     solver.Solve(main_phase, search_log, objective_monitor, limit);
00411   }
