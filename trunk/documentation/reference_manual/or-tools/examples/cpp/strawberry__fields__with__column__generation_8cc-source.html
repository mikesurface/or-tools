00014 // Demonstration of column generation using LP toolkit.
00015 //
00016 // Column generation is the technique of generating columns (aka
00017 // resource bundles aka variables) of the constraint matrix
00018 // incrementally guided by feedback from the constraint duals
00019 // (cost-of-resources).  Frequently this lets one solve large problems
00020 // efficiently, e.g. problems where the number of potential columns is
00021 // exponentially large.
00022 //
00023 // Solves a covering problem taken from ITA Software recruiting web
00024 // site:
00025 //
00026 // "Strawberries are growing in the cells of a rectangular field
00027 // (grid). You want to build greenhouses to enclose the
00028 // strawberries. Greenhouses are rectangular, axis-aligned with the
00029 // field (i.e., not diagonal), and may not overlap. The cost of each
00030 // greenhouse is $10 plus $1 per unit of area covered."
00031 //
00032 // Variables:
00033 //
00034 //    for each Box (greenhouse), continuous variable b{x1,y1,x2,y2} in [0,1]
00035 //
00036 // Constraints:
00037 //
00038 //   box limit:
00039 //     sum b{x1,y1,x2,y2) <= MAX_BOXES
00040 //   non-overlap (for each cell x,y):
00041 //     sum b{x1,y1,x2,y2} <= 1     (summed over containing x1<=x<=x2, y1<=y<=y2)
00042 //   coverage (for each cell x,y with a strawberry):
00043 //     sum b{x1,y1,x2,y2} = 1      (summed over containing x1<=x<=x2, y1<=y<=y2)
00044 //
00045 // Since the number of possible boxes is O(d^4) where d is the linear
00046 // dimension, starts from singleton column (box) covering entire grid,
00047 // ensuring solvability.  Then iteratively the problem is solved and
00048 // the constraint duals (aka reduced costs) used to guide the
00049 // generation of a single new column (box), until convergence or a
00050 // maximum number of iterations.
00051 //
00052 // No attempt is made to force integrality.

00311 class CoveringProblem {
00312  public:
00313   // Grid is a row-major string of length width*height with '@' for an
00314   // occupied cell (strawberry) and '.' for an empty cell.  Solver is
00315   // not owned.
00316   CoveringProblem(MPSolver* const solver, const Instance& instance)
00317       : solver_(solver),
00318         max_boxes_(instance.max_boxes),
00319         width_(instance.width),
00320         height_(instance.height),
00321         grid_(instance.grid) {}
00322 
00323   // Constructs initial variables and constraints.  Initial column
00324   // (box) covers entire grid, ensuring feasibility.
00325   bool Init() {
00326     // Check consistency.
00327     int size = strlen(grid_);
00328     if (size != area()) {
00329       return false;
00330     }
00331     for (int i = 0; i < size; ++i) {
00332       char c = grid_[i];
00333       if ((c != '@') && (c != '.')) return false;
00334     }
00335 
00336     AddCellConstraints();  // sum for every cell is <=1 or =1
00337     AddMaxBoxesConstraint();  // sum of box variables is <= max_boxes()
00338     if (!FLAGS_colgen_complete) {
00339       AddBox(Box(0, width()-1, 0, height()-1));  // grid-covering box
00340     } else {
00341       // Naive alternative to column generation - generate all boxes;
00342       // works fine for smaller problems, too slow for big.
00343       for (int y_min = 0; y_min < height(); ++y_min) {
00344         for (int y_max = y_min; y_max < height(); ++y_max) {
00345           for (int x_min = 0; x_min < width(); ++x_min) {
00346             for (int x_max = x_min; x_max < width(); ++x_max) {
00347               AddBox(Box(x_min, x_max, y_min, y_max));
00348             }
00349           }
00350         }
00351       }
00352     }
00353     return true;
00354   }
00355 
00356   int width() const { return width_; }
00357   int height() const { return height_; }
00358   int area() const { return width() * height(); }
00359   int max_boxes() const { return max_boxes_; }
00360 
00361   bool IsCellOccupied(int x, int y) const { return grid_[index(x, y)] == '@'; }
00362 
00363   // Calculates reduced costs for each possible Box and if any is
00364   // negative (improves cost), returns reduced cost and set target to
00365   // the most-negative (steepest descent) one - otherwise returns 0..
00366   //
00367   // For a problem in standard form 'minimize c*x s.t. Ax<=b, x>=0'
00368   // the reduced cost vector is c - transp(y) * A where y is the dual
00369   // cost column vector.
00370   //
00371   // For this covering problem, in which all coefficients in A are 0
00372   // or 1, this reduces to:
00373   //
00374   //   reduced_cost(box) =
00375   //
00376   //     box.Cost() - sum_{enclosed cell} cell_constraint->dual_value()
00377   //                - max_boxes_constraint_->dual_value()
00378   //
00379   // Since there are O(d^4) boxes, we don't also want O(d^2) sum for
00380   // each, so pre-calculate sums of cell duals for all rectangles with
00381   // upper-left at 0, 0, and use these to calculate the sum in
00382   // constant time using the standard inclusion-exclusion trick.
00383   double GetOptimalBox(Box* const target) {
00384     // Cost change threshold for new Box
00385     const double kCostChangeThreshold = -.01;
00386 
00387     // Precomputes the sum of reduced costs for every upper-left
00388     // rectangle.
00389     std::vector<double> upper_left_sums(area());
00390     ComputeUpperLeftSums(&upper_left_sums);
00391 
00392     const double max_boxes_dual = max_boxes_constraint_->dual_value();
00393     double best_reduced_cost = kCostChangeThreshold;
00394     Box best_box;
00395     for (int y_min = 0; y_min < height(); ++y_min) {
00396       for (int y_max = y_min; y_max < height(); ++y_max) {
00397         for (int x_min = 0; x_min < width(); ++x_min) {
00398           for (int x_max = x_min; x_max < width(); ++x_max) {
00399             Box box(x_min, x_max, y_min, y_max);
00400             const double cell_coverage_dual =  // inclusion-exclusion
00401                 + zero_access(upper_left_sums, x_max, y_max)
00402                 - zero_access(upper_left_sums, x_max, y_min - 1)
00403                 - zero_access(upper_left_sums, x_min - 1, y_max)
00404                 + zero_access(upper_left_sums, x_min - 1, y_min - 1);
00405 
00406             // All coefficients for new column are 1, so no need to
00407             // multiply constraint duals by any coefficients when
00408             // computing the reduced cost.
00409             const double reduced_cost = box.Cost()
00410                 - (cell_coverage_dual + max_boxes_dual);
00411 
00412             if (reduced_cost < best_reduced_cost) {
00413               // Even with negative reduced cost, the box may already
00414               // exist, and even be basic (part of solution)!  This
00415               // counterintuitive situation is due to the problem's
00416               // many redundant linear equality constraints: many
00417               // steepest-edge pivot moves will be of zero-length.
00418               // Ideally one would want to check the length of the
00419               // move but that is difficult without access to the
00420               // internals of the solver (e.g., access to B^-1 in the
00421               // simplex algorithm).
00422               if (boxes_.find(box) == boxes_.end()) {
00423                 best_reduced_cost = reduced_cost;
00424                 best_box = box;
00425               }
00426             }
00427           }
00428         }
00429       }
00430     }
00431 
00432     if (best_reduced_cost < kCostChangeThreshold) {
00433       if (target) {
00434         *target = best_box;
00435       }
00436       return best_reduced_cost;
00437     }
00438     return 0;
00439   }

00441   // Add continuous [0,1] box variable with box.Cost() as objective
00442   // coefficient.  Add to cell constraint of all enclosed cells.
00443   MPVariable* AddBox(const Box& box) {
00444     CHECK(boxes_.find(box) == boxes_.end());
00445     MPVariable* const var = solver_->MakeNumVar(0., 1., box.DebugString());
00446     solver_->SetObjectiveCoefficient(var, box.Cost());
00447     max_boxes_constraint_->SetCoefficient(var, 1.0);
00448     for (int y = box.y_min(); y <= box.y_max(); ++y) {
00449       for (int x = box.x_min(); x <= box.x_max(); ++x) {
00450         cell(x, y)->SetCoefficient(var, 1.0);
00451       }
00452     }
00453     boxes_[box] = var;
00454     return var;
00455   }

00530   // Gets 2d array element, returning 0 if out-of-bounds.
00531   double zero_access(const std::vector<double>& array, int x, int y) const {

00538   // Precomputes the sum of reduced costs for every upper-left
00539   // rectangle.
00540   void ComputeUpperLeftSums(std::vector<double>* upper_left_sums) const {
00541     for (int y = 0; y < height(); ++y) {
00542       for (int x = 0; x < width(); ++x) {
00543         upper_left_sums->operator[](index(x, y)) =
00544             cell(x, y)->dual_value()
00545             + zero_access(*upper_left_sums, x - 1, y)
00546             + zero_access(*upper_left_sums, x, y - 1)
00547             - zero_access(*upper_left_sums, x - 1, y - 1);
00548       }
00549     }
00550   }

00565 // Solves iteratively using delayed column generation, up to maximum
00566 // number of steps.
00567 void SolveInstance(const Instance& instance,
00568                    MPSolver::OptimizationProblemType solver_type) {
00569   // Prepares the solver.
00570   MPSolver solver("ColumnGeneration", solver_type);
00571   solver.SuppressOutput();
00572   solver.SetMinimization();
00573 
00574   // Construct problem.
00575   CoveringProblem problem(&solver, instance);
00576   CHECK(problem.Init());
00577   LOG(INFO) << "Initial problem:\n" << problem.PrintGrid();
00578 
00579   int step_number = 0;
00580   while (step_number < FLAGS_colgen_max_iterations) {
00581     if (FLAGS_colgen_verbose) {
00582       LOG(INFO) << "Step number " << step_number;
00583     }
00584 
00585     // Solve with existing columns.
00586     CHECK_EQ(MPSolver::OPTIMAL, solver.Solve());
00587     if (FLAGS_colgen_verbose) {
00588       LOG(INFO) << problem.PrintCovering();
00589     }
00590 
00591     // Find optimal new column to add, or stop if none.
00592     Box box;
00593     const double reduced_cost = problem.GetOptimalBox(&box);
00594     if (reduced_cost >= 0) {
00595       break;
00596     }
00597 
00598     // Add new column to problem.
00599     if (FLAGS_colgen_verbose) {
00600       LOG(INFO) << "Adding " << box.DebugString()
00601                 << ", reduced_cost =" << reduced_cost;
00602     }
00603     problem.AddBox(box);
00604 
00605     ++step_number;
00606   }
00607 
00608   if (step_number >= FLAGS_colgen_max_iterations) {
00609     // Solve one last time with all generated columns.
00610     CHECK_EQ(MPSolver::OPTIMAL, solver.Solve());
00611   }
00612 
00613   LOG(INFO) << step_number << " columns added";
00614   LOG(INFO) << "Final coverage: " << problem.PrintCovering();
00615 }
00616 }  // namespace operations_research
