00104 void DimacsAssignmentParser<GraphType>::ParseProblemLine(
00105     const char* line) {
00106   static const char* kIncorrectProblemLine =
00107       "Incorrect assignment problem line.";
00108   static const char* kAssignmentProblemType = "asn";
00109   char problem_type[4];
00110   NodeIndex num_nodes;
00111   ArcIndex num_arcs;
00112 
00113   if ((sscanf(line, "%*c%3s%d%d",  // NOLINT
00114               problem_type,
00115               &num_nodes,
00116               &num_arcs) != 3) ||
00117       (strncmp(kAssignmentProblemType,
00118                problem_type,
00119                strlen(kAssignmentProblemType)) != 0)) {
00120     state_.bad = true;
00121     state_.reason = kIncorrectProblemLine;
00122     state_.bad_line.reset(new string(line));
00123     return;
00124   }
00125 
00126   state_.num_arcs = num_arcs;
00127   graph_builder_ = new AnnotatedGraphBuildManager<GraphType>(
00128       num_nodes, num_arcs, FLAGS_assignment_optimize_layout);

00131 template <typename GraphType>
00132 void DimacsAssignmentParser<GraphType>::ParseNodeLine(const char* line) {
00133   NodeIndex node_id;
00134   if (sscanf(line, "%*c%d", &node_id) != 1) {  // NOLINT
00135     state_.bad = true;
00136     state_.reason = "Syntax error in node desciption.";
00137     state_.bad_line.reset(new string(line));
00138     return;
00139   }
00140   if (state_.nodes_described) {
00141     state_.bad = true;
00142     state_.reason = "All node description must precede first arc description.";
00143     state_.bad_line.reset(new string(line));
00144     return;
00145   }
00146   state_.num_left_nodes = ::std::max(state_.num_left_nodes, node_id);

00149 template <typename GraphType>
00150 void DimacsAssignmentParser<GraphType>::ParseArcLine(
00151     const char* line) {
00152   if (graph_builder_ == NULL) {
00153     state_.bad = true;
00154     state_.reason =
00155         "Problem specification line must precede any arc specification.";
00156     state_.bad_line.reset(new string(line));
00157     return;
00158   }
00159   if (!state_.nodes_described) {
00160     state_.nodes_described = true;
00161     DCHECK(assignment_ == NULL);
00162     assignment_ = new LinearSumAssignment<GraphType>(
00163         state_.num_left_nodes, state_.num_arcs);
00164   }
00165   NodeIndex tail;
00166   NodeIndex head;
00167   CostValue cost;
00168   if (sscanf(line, "%*c%d%d%lld", &tail, &head, &cost) != 3) {  // NOLINT
00169     state_.bad = true;
00170     state_.reason = "Syntax error in arc descriptor.";
00171     state_.bad_line.reset(new string(line));
00172   }
00173   ArcIndex arc = graph_builder_->AddArc(tail - 1, head - 1);
00174   assignment_->SetArcCost(arc, FLAGS_assignment_maximize_cost ? -cost : cost);
00175 }

00179 template <typename GraphType>
00180 void DimacsAssignmentParser<GraphType>::ParseOneLine(char* line) {
00181   if (state_.bad) {
00182     return;
00183   }
00184 
00185   if (state_.expect_last_line) {
00186     state_.bad = true;
00187     state_.reason = "Input line is too long.";
00188     // state_.bad_line was already set when we noticed the line
00189     // didn't end with '\n'.
00190     return;
00191   }
00192 
00193   size_t length = strlen(line);
00194   // The final line might not end with newline. Any other line
00195   // that seems not to is actually a line that was too long
00196   // for our input buffer.
00197   if (line[length - 1] != '\n') {
00198     state_.expect_last_line = true;
00199     // Prepare for the worst; we might need to inform the user of
00200     // an error on this line even though we can't detect the error
00201     // yet.
00202     state_.bad_line.reset(new string(line));
00203   }
00204 
00205 
00206   switch (line[0]) {
00207     case 'p': {
00208       // Problem-specification line
00209       ParseProblemLine(line);
00210       break;
00211     }
00212     case 'c': {
00213       // Comment; do nothing.
00214       return;
00215     }
00216     case 'n': {
00217       // Node line defining a node on the left side
00218       ParseNodeLine(line);
00219       break;
00220     }
00221     case 'a': {
00222       ParseArcLine(line);
00223       break;
00224     }
00225     case '0':
00226     case '\n':
00227         break;
00228     default: {
00229       state_.bad = true;
00230       state_.reason = "Unknown line type in the input.";
00231       state_.bad_line.reset(new string(line));
00232       break;
00233     }
00234   }
00235 }

00257 // Reads an assignment problem description from the given file in
00258 // DIMACS format and returns a LinearSumAssignment object representing
00259 // the problem description. For a description of the format, see
00260 // http://lpsolve.sourceforge.net/5.5/DIMACS_asn.htm
00261 //
00262 // Also returns an error message (empty if no error) and a handle on
00263 // the underlying graph representation. The error_message pointer must
00264 // not be NULL because we insist on returning an explanatory message
00265 // in the case of error. The graph_handle pointer must not be NULL
00266 // because unless we pass a non-const pointer to the graph
00267 // representation back to the caller, the caller lacks a good way to
00268 // free the underlying graph (which isn't owned by the
00269 // LinearAssignment instance).
00270 template <typename GraphType>
00271 LinearSumAssignment<GraphType>* DimacsAssignmentParser<GraphType>::Parse(
00272     string* error_message,
00273     GraphType** graph_handle) {
00274   CHECK_NOTNULL(error_message);
00275   CHECK_NOTNULL(graph_handle);
00276   Callback1<char*>* cb =
00277       NewPermanentCallback(
00278           this, &DimacsAssignmentParser<GraphType>::ParseOneLine);
00279   // ParseFileByLines takes ownership of cb and deletes it.
00280   ParseFileByLines(filename_, cb);
00281 
00282   if (state_.bad) {
00283     *error_message = state_.reason;
00284     *error_message = *error_message + ": \"" + *state_.bad_line + "\"";
00285     return NULL;
00286   }
00287   if (graph_builder_ == NULL) {
00288     *error_message = "empty graph description";
00289     return NULL;
00290   }
00291   scoped_ptr<PermutationCycleHandler<ArcIndex> > cycle_handler(
00292       assignment_->ArcAnnotationCycleHandler());
00293   GraphType* graph = graph_builder_->Graph(cycle_handler.get());
00294   if (graph == NULL) {
00295     *error_message = "unable to create compact static graph";
00296     return NULL;
00297   }
00298   assignment_->SetGraph(graph);
00299   *error_message = "";
00300   // Return a handle on the graph to the caller so the caller can free
00301   // the graph's memory, because the LinearSumAssignment object does
00302   // not take ownership of the graph and hence will not free it.
00303   *graph_handle = graph;
00304   return assignment_;
00305 }
