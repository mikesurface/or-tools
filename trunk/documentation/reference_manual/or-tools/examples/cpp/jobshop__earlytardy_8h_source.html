00014 // This model implements a simple jobshop problem with
00015 // earlyness-tardiness costs.
00016 //
00017 // A earlyness-tardinessjobshop is a standard scheduling problem where
00018 // you must schedule a set of jobs on a set of machines.  Each job is
00019 // a sequence of tasks (a task can only start when the preceding task
00020 // finished), each of which occupies a single specific machine during
00021 // a specific duration. Therefore, a job is a sequence of pairs
00022 // (machine id, duration), along with a release data (minimum start
00023 // date of the first task of the job, and due data (end time of the
00024 // last job) with a tardiness linear penalty.
00025 
00026 // The objective is to minimize the sum of early-tardy penalties for each job.
00027 //
00028 // This will be modelled by sets of intervals variables (see class
00029 // IntervalVar in constraint_solver/constraint_solver.h), one per
00030 // task, representing the [start_time, end_time] of the task.  Tasks
00031 // in the same job will be linked by precedence constraints.  Tasks on
00032 // the same machine will be covered by Sequence constraints.
00033

00094   void GenerateRandomData(int machine_count,
00095                           int job_count,
00096                           int max_release_date,
00097                           int max_early_cost,
00098                           int max_tardy_cost,
00099                           int max_duration,
00100                           int scale_factor,
00101                           int seed) {
00102     name_ = StringPrintf("EtJobshop(m%d-j%d-mrd%d-mew%d-mtw%d-md%d-sf%d-s%d)",
00103                          machine_count,
00104                          job_count,
00105                          max_release_date,
00106                          max_early_cost,
00107                          max_tardy_cost,
00108                          max_duration,
00109                          scale_factor,
00110                          seed);
00111     LOG(INFO) << "Generating random problem " << name_;
00112     ACMRandom random(seed);
00113     machine_count_ = machine_count;
00114     job_count_ = job_count;
00115     for (int j = 0; j < job_count_; ++j) {
00116       const int release_date = random.Uniform(max_release_date);
00117       int sum_of_durations = max_release_date;
00118       all_jobs_.push_back(Job(release_date,
00119                               0,  // due date, to be filled later.
00120                               random.Uniform(max_early_cost),
00121                               random.Uniform(max_tardy_cost)));
00122       for (int m = 0; m < machine_count_; ++m) {
00123         const int duration = random.Uniform(max_duration);
00124         all_jobs_.back().all_tasks.push_back(Task(j, m, duration));
00125         sum_of_durations += duration;
00126       }
00127       all_jobs_.back().due_date = sum_of_durations * scale_factor / 100;
00128       horizon_ += all_jobs_.back().due_date;
00129       // Scramble jobs.
00130       for (int m = 0; m < machine_count_; ++m) {
00131         Task t = all_jobs_.back().all_tasks[m];
00132         const int target = random.Uniform(machine_count_);
00133         all_jobs_.back().all_tasks[m] = all_jobs_.back().all_tasks[target];
00134         all_jobs_.back().all_tasks[target] = t;
00135       }
00136     }
00137   }

00158   void ProcessNewJetLine(char* const line) {
00159     // TODO(user): more robust logic to support single-task jobs.
00160     static const char kWordDelimiters[] = " ";
00161     std::vector<string> words;
00162     SplitStringUsing(line, kWordDelimiters, &words);
00163     if (words.size() == 2) {
00164       job_count_ = atoi32(words[0]);
00165       machine_count_ = atoi32(words[1]);
00166       CHECK_GT(machine_count_, 0);
00167       CHECK_GT(job_count_, 0);
00168       LOG(INFO) << machine_count_ << "  - machines and "
00169                 << job_count_ << " jobs";
00170     } else if (words.size() > 2 && machine_count_ != 0) {
00171       const int job_id = all_jobs_.size();
00172       CHECK_EQ(words.size(), machine_count_ * 2 + 3);
00173       const int due_date = atoi32(words[2 * machine_count_]);
00174       const int early_cost = atoi32(words[2 * machine_count_ + 1]);
00175       const int late_cost = atoi32(words[2 * machine_count_ + 2]);
00176       LOG(INFO) << "Add job with due date = " << due_date
00177                 << ", early cost = " << early_cost
00178                 << ", and late cost = " << late_cost;
00179       all_jobs_.push_back(Job(0, due_date, early_cost, late_cost));
00180       for (int i = 0; i < machine_count_; ++i) {
00181         const int machine_id = atoi32(words[2 * i]);
00182         const int duration = atoi32(words[2 * i + 1]);
00183         all_jobs_.back().all_tasks.push_back(
00184             Task(job_id, machine_id, duration));
00185         horizon_ += duration;
00186       }
00187     }
00188   }
