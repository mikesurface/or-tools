00027 // A monoid is an algebraic structure consisting of a set S with an associative
00028 // binary operation * :S x S -> S that has an identity element.
00029 // Associative means a*(b*c) = (a*b)*c for all a,b,c in S.
00030 // An identity element is an element e in S such that for all a in S,
00031 // e*a = a*e = a.
00032 // See http://en.wikipedia.org/wiki/Monoid for more details.
00033 //
00034 // A MonoidOperationTree is a data structure that maintains a product
00035 // a_1 * a_2 * ... * a_n for a given (fixed) n, and that supports the
00036 // following functions:
00037 // - Setting the k-th operand to a given value in O(log n) calls to the *
00038 // operation
00039 // - Querying the result in O(1)
00040 //
00041 // Note that the monoid is not required to be commutative.
00042 //
00043 // The parameter class T represents an element of the set S.
00044 // It must:
00045 //   * Have a public no-argument constructor producing the identity element.
00046 //   * Have a Set(const T& value) method that sets its value to the given one.
00047 //   * Have a Compute(const T& left, const T& right) method that sets its value
00048 //        to the result of the binary operation for the two given operands.
00049 //   * Have a string DebugString() const method.
00050 //
00051 // Possible use cases are:
00052 // * Maintain a sum or a product of doubles, with a guarantee that the queried
00053 //       result is independent of the order of past numerical issues
00054 // * Maintain a product of identically sized square matrices, which is an
00055 //       example of use with non-commutative operations.
00056 template<class T>
00057 class MonoidOperationTree {
00058  public:
00059   // Constructs a MonoidOperationTree able to store 'size' operands.
00060   explicit MonoidOperationTree(int size);
00061   virtual ~MonoidOperationTree() {}
00062 
00063   // Returns the root of the tree, containing the result of the operation.
00064   const T& result() const { return *result_; }
00065 
00066   // Resets the argument of given index.
00067   void Reset(int argument_index);
00068 
00069   // Sets the argument of given index.
00070   void Set(int argument_index, const T& argument);
00071 
00072   // Resets all arguments.
00073   void Clear();
00074 
00075   // Returns the leaf node corresponding to the given argument index.
00076   const T& GetOperand(int argument_index) const {

00080   // Dive down a branch of the operation tree, and then come back up.

00088  private:
00089   // Computes the index of the first leaf for the given size.
00090   static int ComputeLeafOffset(int size);
00091 
00092   // Computes the total number of nodes we need to store non-leaf nodes and
00093   // leaf nodes.
00094   static int ComputeNumberOfNodes(int leaf_offset);
00095 
00096   // Computes the whole path from the node of given position up to the root,
00097   // excluding the bottom node.
00098   void ComputeAbove(int position);
00099 
00100   // Computes the node of given position, and no other.
00101   void Compute(int position);
00102 
00103   // Returns the position of the leaf node of given index.
00104   int PositionOfLeaf(int index) const { return leaf_offset_ + index; }
00105 
00106   // Returns true if the node of given position is a leaf.
00107   bool IsLeaf(int position) const { return position >= leaf_offset_; }
00108 
00109   // Returns the index of the argument stored in the node of given position.
00110   int ArgumentIndexOfLeafPosition(int position) const {
00111     DCHECK(IsLeaf(position));

00113   }

00118   static int father(int pos) { return (pos - 1) >> 1; }

00120   static int right(int pos) { return (pos + 1) << 1; }

00122   // The number of arguments that can be stored in this tree. That is, the
00123   // number of used leaves. (There may be unused leaves, too)
00124   const int size_;
00125 
00126   // The index of the first leaf.
00127   const int leaf_offset_;
00128 
00129   // Number of nodes, both non-leaves and leaves.
00130   const int num_nodes_;
00131 
00132   // All the nodes, both non-leaves and leaves.
00133   scoped_array<T> nodes_;
00134 
00135   // A pointer to the root node

00137 
00138   // The identity element
00139   const T identity_;

00148 template<class T>
00149 int MonoidOperationTree<T>::ComputeLeafOffset(int size) {
00150   int smallest_pow_two_not_less_than_size = 1;
00151   while (smallest_pow_two_not_less_than_size < size) {
00152     smallest_pow_two_not_less_than_size <<= 1;
00153   }
00154   return std::max(1, smallest_pow_two_not_less_than_size - 1);
00155 }

00157 template<class T>
00158 int MonoidOperationTree<T>::ComputeNumberOfNodes(int leaf_offset) {
00159   // leaf_offset should be a power of 2 minus 1.
00160   DCHECK_EQ(0, (leaf_offset) & (leaf_offset+1));
00161   const int num_leaves = leaf_offset + 1;
00162   const int num_nodes = leaf_offset + num_leaves;
00163   DCHECK_GE(num_nodes, 3);  // We need at least the root and its 2 children
00164   return num_nodes;
00165 }

00204 template<class T>
00205 void MonoidOperationTree<T>::ComputeAbove(int position) {
00206   int pos = father(position);
00207   while (pos > 0) {
00208     Compute(pos);
00209     pos = father(pos);
00210   }
00211   Compute(0);
00212 }

00215 void MonoidOperationTree<T>::Compute(int position) {
00216   const T& left_child = nodes_[left(position)];
00217   const T& right_child = nodes_[right(position)];
00218   nodes_[position].Compute(left_child, right_child);

00237 template <class T>
00238 template <class Diver>
00239 void MonoidOperationTree<T>::DiveInTree(int position, Diver* diver) const {
00240   // Are we at a leaf?
00241   if (IsLeaf(position)) {
00242     const int index = ArgumentIndexOfLeafPosition(position);
00243     const T& argument = nodes_[position];
00244     diver->OnArgumentReached(index, argument);
00245   } else {
00246     const T& current = nodes_[position];
00247     const T& left_child = nodes_[left(position)];
00248     const T& right_child = nodes_[right(position)];
00249     if (diver->ChooseGoLeft(current, left_child, right_child)) {
00250       // Go left
00251       DiveInTree(left(position), diver);
00252       // Come back up
00253       diver->OnComeBackFromLeft(current, left_child, right_child);
00254     } else {
00255       // Go right
00256       DiveInTree(right(position), diver);
00257       // Come back up
00258       diver->OnComeBackFromRight(current, left_child, right_child);
00259     }
00260   }
00261 }
