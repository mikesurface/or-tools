00025 // This class stores a vector of distinct elements, as well as a map
00026 // from elements to index to find the index in the vector.
00027 // This is useful to store mapping between objects and indices.
00028 template <class T> class VectorMap {
00029  public:
00030   // Adds an element if not already present, and returns its index in
00031   // the vector-map.
00032   int Add(const T& element) {
00033     int current_index = Index(element);
00034     if (current_index != -1) {
00035       return current_index;
00036     }
00037     const int index = list_.size();
00038     CHECK_EQ(index, map_.size());
00039     list_.push_back(element);
00040     map_[element] = index;
00041     return index;
00042   }
00043   // TODO(user): Use ArraySlice.
00044 
00045   // Adds all elements of the vector.
00046   void Add(const std::vector<T>& elements) {
00047     for (int i = 0; i < elements.size(); ++i) {
00048       Add(elements[i]);
00049     }
00050   }
00051 
00052   // Returns -1 if the element is not in the vector, or its unique
00053   // index if it is.
00054   int Index(const T& element) const {
00055     return FindWithDefault(map_, element, -1);
00056   }
00057   // TODO(user): explore a int-type version.
00058 
00059   // Returns wether the element has already been added to the vector-map.
00060   bool Contains(const T& element) const {
00061     return ContainsKey(map_, element);
00062   }
00063 
00064   // Returns the element at position index.
00065   const T& Element(int index) const {
00066     CHECK_GE(index, 0);
00067     CHECK_LT(index, list_.size());
00068     return list_[index];
00069   }
00070 
00071   // Returns the number of distinct elements added to the vector-map.
00072   int size() const { return list_.size(); }
00073 
00074   // Clears all the elements added to the vector-map.
00075   void clear() {
00076     list_.clear();
00077     map_.clear();
00078   }
00079 
00080   // Returns a read-only access to the vector of elements.
00081   const std::vector<T>& list() const { return list_; }
