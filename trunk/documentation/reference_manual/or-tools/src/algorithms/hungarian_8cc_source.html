00300   // Find a maximal assignment by subtracting each of the

00519 // Step 3.
00520 // Cover each column containing a starred zero.  If all columns are
00521 // covered, the starred zeros describe a complete set of unique assignments.
00522 // In this case, terminate the algorithm.  Otherwise, go to step 4.

00524   int num_covered = 0;
00525 
00526   for (int col = 0; col < matrix_size_; ++col) {
00527     if (ColContainsStar(col)) {
00528       CoverCol(col);
00529       num_covered++;
00530     }
00531   }

00540 // Step 4.
00541 // Find a noncovered zero and prime it.  If there is no starred zero in the
00542 // row containing this primed zero, Go to Step 5.  Otherwise, cover this row
00543 // and uncover the column containing the starred zero. Continue in this manner
00544 // until there are no uncovered zeros left, then go to Step 6.
00545 
00546 void HungarianOptimizer::PrimeZeroes() {
00547   // This loop is guaranteed to terminate in at most matrix_size_ iterations,
00548   // as findZero() returns a location only if there is at least one uncovered
00549   // zero in the matrix.  Each iteration, either one row is covered or the
00550   // loop terminates.  Since there are matrix_size_ rows, after that many
00551   // iterations there are no uncovered cells and hence no uncovered zeroes,
00552   // so the loop terminates.
00553   for (;;) {
00554     int zero_row, zero_col;
00555     if (!FindZero(&zero_row, &zero_col)) {
00556       // No uncovered zeroes.
00557       state_ = &HungarianOptimizer::AugmentPath;
00558       return;
00559     }
00560 
00561     Prime(zero_row, zero_col);
00562     int star_col = FindStarInRow(zero_row);
00563 
00564     if (star_col != kHungarianOptimizerColNotFound) {
00565       CoverRow(zero_row);
00566       UncoverCol(star_col);
00567     } else {
00568       preimage_[0] = zero_row;
00569       image_[0] = zero_col;
00570       state_ = &HungarianOptimizer::MakeAugmentingPath;
00571       return;
00572     }
00573   }
00574 }

00576 // Step 5.
00577 // Construct a series of alternating primed and starred zeros as follows.
00578 // Let Z0 represent the uncovered primed zero found in Step 4.  Let Z1 denote
00579 // the starred zero in the column of Z0 (if any). Let Z2 denote the primed
00580 // zero in the row of Z1 (there will always be one).  Continue until the
00581 // series terminates at a primed zero that has no starred zero in its column.
00582 // Unstar each starred zero of the series, star each primed zero of the
00583 // series, erase all primes and uncover every line in the matrix.  Return to
00584 // Step 3.
00585 void HungarianOptimizer::MakeAugmentingPath() {
00586   bool done = false;
00587   int count = 0;
00588 
00589   // Note: this loop is guaranteed to terminate within matrix_size_ iterations
00590   // because:
00591   // 1) on entry to this step, there is at least 1 column with no starred zero
00592   //    (otherwise we would have terminated the algorithm already.)
00593   // 2) each row containing a star also contains exactly one primed zero.
00594   // 4) each column contains at most one starred zero.
00595   //
00596   // Since the path_ we construct visits primed and starred zeroes alternately,
00597   // and terminates if we reach a primed zero in a column with no star, our
00598   // path_ must either contain matrix_size_ or fewer stars (in which case the
00599   // loop iterates fewer than matrix_size_ times), or it contains more.  In
00600   // that case, because (1) implies that there are fewer than
00601   // matrix_size_ stars, we must have visited at least one star more than once.
00602   // Consider the first such star that we visit more than once; it must have
00603   // been reached immediately after visiting a prime in the same row.  By (2),
00604   // this prime is unique and so must have also been visited more than once.
00605   // Therefore, that prime must be in the same column as a star that has been
00606   // visited more than once, contradicting the assumption that we chose the
00607   // first multiply visited star, or it must be in the same column as more
00608   // than one star, contradicting (3).  Therefore, we never visit any star
00609   // more than once and the loop terminates within matrix_size_ iterations.
00610 
00611   while (!done) {
00612     // First construct the alternating path...
00613     int row = FindStarInCol(image_[count]);
00614 
00615     if (row != kHungarianOptimizerRowNotFound) {
00616       count++;
00617       preimage_[count] = row;
00618       image_[count] = image_[count-1];
00619     } else {
00620       done = true;
00621     }
00622 
00623     if (!done) {
00624       int col = FindPrimeInRow(preimage_[count]);
00625       count++;
00626       preimage_[count] = preimage_[count-1];
00627       image_[count] = col;
00628     }
00629   }
00630 
00631   // Then modify it.
00632   for (int i = 0; i <= count; ++i) {
00633     int row = preimage_[i];
00634     int col = image_[i];
00635 
00636     if (IsStarred(row, col)) {
00637       UnStar(row, col);
00638     } else {
00639       Star(row, col);
00640     }
00641   }
00642 
00643   ClearCovers();
00644   ClearPrimes();

00648 // Step 6
00649 // Add the smallest uncovered value in the matrix to every element of each
00650 // covered row, and subtract it from every element of each uncovered column.
00651 // Return to Step 4 without altering any stars, primes, or covered lines.
00652 void HungarianOptimizer::AugmentPath() {
00653   double minval = FindSmallestUncovered();
00654 
00655   for (int row = 0; row < matrix_size_; ++row) {
00656     for (int col = 0; col < matrix_size_; ++col) {
00657       if (RowCovered(row)) {
00658         costs_[row][col] += minval;
00659       }
00660 
00661       if (!ColCovered(col)) {
00662         costs_[row][col] -= minval;
00663       }
00664     }
00665   }
00666 
00667   state_ = &HungarianOptimizer::PrimeZeroes;
00668 }
