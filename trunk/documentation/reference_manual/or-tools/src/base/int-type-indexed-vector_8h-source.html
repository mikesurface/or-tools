00014 // This file provides the ITIVector container that wraps around the STL vector.
00015 // The wrapper restrict indexing to a pre-specified type-safe integer type or
00016 // IntType (see base/int-type.h).  It prevents accidental indexing
00017 // by different "logical" integer-like types (e.g.  another IntType) or native
00018 // integer types.  The wrapper is useful as C++ and the standard template
00019 // library allows the user to mix "logical" integral indices that might have a
00020 // different role.
00021 //
00022 // The container can only be indexed by an instance of an IntType class, which
00023 // can be declared as:
00024 //
00025 //     DEFINE_INT_TYPE(IntTypeName, IntTypeValueType);
00026 //
00027 // where IntTypeName is the desired name for the "logical" integer-like type
00028 // and the ValueType is a supported native integer type such as int or
00029 // uint64 (see base/int-type.h for details).
00030 //
00031 // The wrapper exposes all public methods of STL vector and behaves mostly as
00032 // pass-through.  The only method modified to ensure type-safety is the operator
00033 // [] and the at() method.
00034 //
00035 // EXAMPLES --------------------------------------------------------------------
00036 //
00037 //    DEFINE_INT_TYPE(PhysicalChildIndex, int32);
00038 //    ITIVector<PhysicalChildIndex, ChildStats*> vec;
00039 //
00040 //    PhysicalChildIndex physical_index;
00041 //    vec[physical_index] = ...;        <-- index type match: compiles properly.
00042 //    vec.at(physical_index) = ...;     <-- index type match: compiles properly.
00043 //
00044 //    int32 physical_index;
00045 //    vec[physical_index] = ...;        <-- fails to compile.
00046 //    vec.at(physical_index) = ...;     <-- fails to compile.
00047 //
00048 //    DEFINE_INT_TYPE(LogicalChildIndex, int32);
00049 //    int32 logical_index;
00050 //    vec[logical_index] = ...;        <-- fails to compile.
00051 //    vec.at(logical_index) = ...;     <-- fails to compile.
00052 //
00053 // NB: Iterator arithmetic is not allowed as the iterators are not wrapped
00054 // themselves.  Therefore, the following caveat is possible:
00055 //    *(vec.begin() + 0) = ...;
00056
