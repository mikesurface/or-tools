00026 template<typename T> class AdjustablePriorityQueue {
00027  public:
00028   AdjustablePriorityQueue() {}
00029 
00030   void Add(T* const val) {
00031     elems_.push_back(val);
00032     AdjustUpwards(elems_.size() - 1);
00033   }
00034 
00035   void Remove(T* const val) {
00036     const int i = val->GetHeapIndex();
00037     if (i == elems_.size() - 1) {
00038       elems_.pop_back();
00039       return;
00040     }
00041     elems_[i] = elems_.back();
00042     elems_[i]->SetHeapIndex(i);
00043     elems_.pop_back();
00044     NoteChangedPriority(elems_[i]);
00045   }
00046 
00047   bool Contains(const T* const val) const {
00048     const int i = val->GetHeapIndex();
00049     if (i < 0 || i >= elems_.size() || elems_[i] != val) {
00050       return false;
00051     }
00052     return true;
00053   }
00054 
00055   void NoteChangedPriority(T* val) {
00056     const int i = val->GetHeapIndex();
00057     const int parent = (i - 1) / 2;
00058     if (*elems_[parent] < *val) {
00059       AdjustUpwards(i);
00060     } else {
00061       AdjustDownwards(i);
00062     }
00063   }
00064 
00065   T* Top() { return elems_[0]; }
00066 
00067   const T* Top() const { return elems_[0]; }
00068 
00069   void Pop() { Remove(Top()); }
00070 
00071   int Size() const { return elems_.size(); }
00072 
00073   bool IsEmpty() const { return elems_.empty(); }
00074 
00075   void Clear() { elems_.clear(); }
00076 
00077   void CheckValid() const {
00078     for (int i = 0; i < elems_.size(); ++i) {
00079       const int left_child = 1 + 2 * i;
00080       if (left_child < elems_.size()) {
00081         CHECK_GE(elems_[i], elems_[left_child]);
00082       }
00083       const int right_child = left_child + 1;
00084       if (right_child < elems_.size()) {
00085         CHECK_GE(elems_[i], elems_[right_child]);
00086       }
00087     }
00088   }

00090  private:
00091   void AdjustUpwards(int i) {
00092     T* const t = elems_[i];
00093     while (i > 0) {
00094       const int parent = (i - 1) / 2;
00095       if (!(*elems_[parent] < *t)) {
00096         break;
00097       }
00098       elems_[i] = elems_[parent];
00099       elems_[i]->SetHeapIndex(i);
00100       i = parent;
00101     }
00102     elems_[i] = t;
00103     t->SetHeapIndex(i);
00104   }
00105 
00106   void AdjustDownwards(int i) {
00107     T* const t = elems_[i];
00108     while (true) {
00109       const int left_child = 1 + 2 * i;
00110       if (left_child >= elems_.size()) {
00111         break;
00112       }
00113       const int right_child = left_child + 1;
00114       const int next_i = (right_child < elems_.size() &&
00115                           *elems_[left_child] < *elems_[right_child]) ?
00116           right_child :
00117           left_child;
00118       if (!(*t < *elems_[next_i])) {
00119         break;
00120       }
00121       elems_[i] = elems_[next_i];
00122       elems_[i]->SetHeapIndex(i);
00123       i = next_i;
00124     }
00125     elems_[i] = t;
00126     t->SetHeapIndex(i);
00127   }
