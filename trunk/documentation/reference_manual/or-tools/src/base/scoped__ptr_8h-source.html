00039 // A scoped_ptr<T> is like a T*, except that the destructor of scoped_ptr<T>
00040 // automatically deletes the pointer it holds (if any).
00041 // That is, scoped_ptr<T> owns the T object that it points to.
00042 // Like a T*, a scoped_ptr<T> may hold either NULL or a pointer to a T object.
00043 //
00044 // The size of a scoped_ptr is small:
00045 // sizeof(scoped_ptr<C>) == sizeof(C*)
00046 template <class C>
00047 class scoped_ptr {
00048  public:
00049 
00050   // The element type
00051   typedef C element_type;
00052 
00053   // Constructor.  Defaults to intializing with NULL.
00054   // There is no way to create an uninitialized scoped_ptr.
00055   // The input parameter must be allocated with new.
00056   explicit scoped_ptr(C* p = NULL) : ptr_(p) { }
00057 
00058   // Destructor.  If there is a C object, delete it.
00059   // We don't need to test ptr_ == NULL because C++ does that for us.
00060   ~scoped_ptr() {
00061     enum { type_must_be_complete = sizeof(C) };
00062     delete ptr_;
00063   }
00064 
00065   // Reset.  Deletes the current owned object, if any.
00066   // Then takes ownership of a new object, if given.
00067   // this->reset(this->get()) works.
00068   void reset(C* p = NULL) {
00069     if (p != ptr_) {
00070       enum { type_must_be_complete = sizeof(C) };
00071       delete ptr_;
00072       ptr_ = p;
00073     }
00074   }
00075 
00076   // Accessors to get the owned object.
00077   // operator* and operator-> will assert() if there is no current object.
00078   C& operator*() const {
00079     assert(ptr_ != NULL);
00080     return *ptr_;
00081   }
00082   C* operator->() const  {
00083     assert(ptr_ != NULL);
00084     return ptr_;
00085   }
00086   C* get() const { return ptr_; }
00087 
00088   // Comparison operators.
00089   // These return whether two scoped_ptr refer to the same object, not just to
00090   // two different but equal objects.
00091   bool operator==(C* p) const { return ptr_ == p; }
00092   bool operator!=(C* p) const { return ptr_ != p; }
00093 
00094   // Swap two scoped pointers.
00095   void swap(scoped_ptr& p2) {
00096     C* tmp = ptr_;
00097     ptr_ = p2.ptr_;
00098     p2.ptr_ = tmp;
00099   }
00100 
00101   // Release a pointer.
00102   // The return value is the current pointer held by this object.
00103   // If this object holds a NULL pointer, the return value is NULL.
00104   // After this operation, this object will hold a NULL pointer,
00105   // and will not own the object any more.
00106   C* release() {
00107     C* retVal = ptr_;
00108     ptr_ = NULL;
00109     return retVal;
00110   }
00111 
00112  private:
00113   C* ptr_;
00114 
00115   // friend class that can access copy ctor (although if it actually
00116   // calls a copy ctor, there will be a problem) see below
00117   friend scoped_ptr<C> make_scoped_ptr<C>(C *p);
00118 
00119   // Forbid comparison of scoped_ptr types.  If C2 != C, it totally doesn't
00120   // make sense, and if C2 == C, it still doesn't make sense because you should
00121   // never have the same object owned by two different scoped_ptrs.
00122   template <class C2> bool operator==(scoped_ptr<C2> const& p2) const;
00123   template <class C2> bool operator!=(scoped_ptr<C2> const& p2) const;
00124 
00125   DISALLOW_COPY_AND_ASSIGN(scoped_ptr);
00126 };

00144 template <class C>
00145 scoped_ptr<C> make_scoped_ptr(C *p) {
00146   // This does nothing but to return a scoped_ptr of the type that the passed
00147   // pointer is of.  (This eliminates the need to specify the name of T when
00148   // making a scoped_ptr that is used anonymously/temporarily.)  From an
00149   // access control point of view, we construct an unnamed scoped_ptr here
00150   // which we return and thus copy-construct.  Hence, we need to have access
00151   // to scoped_ptr::scoped_ptr(scoped_ptr const &).  However, it is guaranteed
00152   // that we never actually call the copy constructor, which is a good thing
00153   // as we would call the temporary's object destructor (and thus delete p)
00154   // if we actually did copy some object, here.
00155   return scoped_ptr<C>(p);
00156 }
